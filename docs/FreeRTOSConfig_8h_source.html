39/* ── STM32F411xE memory map (mirrors stm32_f411_FLASH.ld) ────────────────── */
40/*
41 * Flash 512 K total
42 * FLASH_BOOT 0x0800_0000 32 K sectors 0–1 (bootloader)
43 * FLASH_OS 0x0800_8000 224 K sectors 2–5 (OS + kernel + drivers)
44 * FLASH_APP 0x0804_0000 256 K sector 6–7 (application)
45 *
46 * RAM 128 K total (0x2000_0000 – 0x2001_FFFF)
47 * FreeRTOS heap 80 000 B (~78 K) heap_4 static array in .bss
48 * .data / .bss ~ 49 KB globals, handles, ring buffers
49 * MSP stack 1 024 B linker _Min_Stack_Size (grows downward from _estack)
50 * Newlib heap 512 B linker _Min_Heap_Size (sbrk region, not RTOS heap)
51 */

54#define configFLASH_OS_ORIGIN 0x08008000UL
