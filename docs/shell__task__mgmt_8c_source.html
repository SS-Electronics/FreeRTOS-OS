4 * This file is part of FreeRTOS-OS Project.

6 * heap
7 * ────
8 * Queries FreeRTOS heap APIs directly for live accuracy and pulls
9 * malloc_fail_count from the task-manager health snapshot.
10 *
11 * Example output:
12 * Heap usage:
13 * Total : 49152 B
14 * Used : 14320 B
15 * Free : 34832 B
16 * MinFree: 33104 B (min ever free)
17 * Errors : 0 malloc failure(s)
18 *
19 * tasks
20 * ─────
21 * Reads from the task-manager health snapshot (refreshed every 5 s).
22 * Uses FreeRTOS+CLI multi-shot output (returns pdTRUE per line) so the
23 * output is not limited by SHELL_OUT_BUF_LEN.
24 *
25 * Example output:
26 * Tasks (5 registered)
27 * ID Name State StackHWM
28 * ----------------------------------------
29 * 1 task_mgr blocked 128 w
30 * 2 uart_mgmt blocked 256 w
31 * [T] Tmr Svc blocked 256 w (timer daemon)

88/* ── heap command ─────────────────────────────────────────────────────────── */
89
90static BaseType_t _cmd_heap_fn(char *out, size_t len, const char *in)
91{
92 (void)in;
93
94 const sys_health_t *h = task_mgr_get_health();
95 size_t total = (size_t)configTOTAL_HEAP_SIZE;
96 size_t free = (size_t)xPortGetFreeHeapSize();
97 size_t used = total - free;
98 size_t minf = (size_t)xPortGetMinimumEverFreeHeapSize();
99 uint32_t faults = h ? h->malloc_fail_count : 0U;
100
101 snprintf(out, len,
102 "Heap usage:\r\n"
103 " Total : %6u B\r\n"
104 " Used : %6u B\r\n"
105 " Free : %6u B\r\n"
106 " MinFree: %6u B (min ever free)\r\n"
107 " Errors : %lu malloc failure(s)\r\n",
108 (unsigned)total,
109 (unsigned)used,
110 (unsigned)free,
111 (unsigned)minf,
112 (unsigned long)faults);
113
114 return pdFALSE;
115}

117/* ── tasks command (multi-shot) ───────────────────────────────────────────── */
118
119/*
120 * Multi-shot state: -1 = header not sent yet, 0..N = next task line index,
121 * TASK_MGR_MAX_TASKS + 1 = timer-daemon line, TASK_MGR_MAX_TASKS + 2 = summary.
122 * Reset to -1 after the final pdFALSE return so a repeated 'tasks' command
123 * starts fresh.
124 */
125static int16_t _tasks_line = -1;
126
127static BaseType_t _cmd_tasks_fn(char *out, size_t len, const char *in)
128{
129 (void)in;
130
131 const sys_health_t *h = task_mgr_get_health();
132
133 if (h == NULL)
134 {
135 _tasks_line = -1;
136 snprintf(out, len, "Task manager not available.\r\n");
137 return pdFALSE;
138 }
139
140 /* ── Header ── */
141 if (_tasks_line == -1)
142 {
143 snprintf(out, len,
144 "Tasks (%u registered)\r\n"
145 " ID %-16s %-10s StackHWM\r\n"
146 " ----------------------------------------\r\n",
147 (unsigned)h->task_count,
148 "Name", "State");
149 _tasks_line = 0;
150 return pdTRUE;
151 }
152
153 /* ── Per-task lines ── */
154 if (_tasks_line < (int16_t)h->task_count)
155 {
156 const task_health_t *t = &h->tasks[_tasks_line];
157 snprintf(out, len,
158 " %2lu %-16s %-10s %u w\r\n",
159 (unsigned long)t->thread_id,
160 t->name ? t->name : "?",
161 _state_to_str(t->state),
162 (unsigned)t->stack_hwm);
163 _tasks_line++;
164 return pdTRUE;
165 }
166
167 /* ── Timer-daemon line ── */
168 if (_tasks_line == (int16_t)h->task_count && h->timer_task_valid)
169 {
170 const task_health_t *t = &h->timer_task;
171 snprintf(out, len,
172 " [T] %-16s %-10s %u w (timer daemon)\r\n",
173 t->name ? t->name : "Tmr Svc",
174 _state_to_str(t->state),
175 (unsigned)t->stack_hwm);
176 _tasks_line++;
177 return pdTRUE;
178 }
179
180 /* ── Summary / final line ── */
181 snprintf(out, len,
182 " Stack overflow events: %lu\r\n",
183 (unsigned long)h->stack_overflow_count);
184 _tasks_line = -1;
185 return pdFALSE;
186}

188/* ── debug command ────────────────────────────────────────────────────────── */
189
190static BaseType_t _cmd_debug_fn(char *out, size_t len, const char *in)
191{
192 BaseType_t plen;
193 const char *param = FreeRTOS_CLIGetParameter(in, 1, &plen);
194
195 if (param == NULL)
196 {
197 snprintf(out, len, "Usage: debug <en|dis>\r\n");
198 return pdFALSE;
199 }
200
201 if (plen == 2 && strncmp(param, "en", 2) == 0)
202 {
203 printk_enable();
204 snprintf(out, len, "Debug output enabled.\r\n");
205 }
206 else if (plen == 3 && strncmp(param, "dis", 3) == 0)
207 {
208 printk_disable();
209 snprintf(out, len, "Debug output disabled.\r\n");
210 }
211 else
212 {
213 snprintf(out, len, "Unknown option '%.*s'. Usage: debug <en|dis>\r\n",
214 (int)plen, param);
215 }
216
217 return pdFALSE;
218}
