51/**
52 * @typedef thread_func_t
53 * @brief Function pointer type for generic threads.
54 * @param parm a container pointer that will be passed
55 * to the thread when schedular starts it
56 *
57 * This type is used to define the body of a thread.
58 * The function must match the signature: void function(void* parm)
59 * @warning The function should never return
60 */

64/**
65 * @struct thread_handle_t
66 * @brief Holds handles, id, priority and parameter data of a thread.
67 * @note Intrusive list node (list) follows the Linux kernel pattern:
68 * embed struct list_node inside the object so the generic list
69 * primitives work without extra allocation.
70 */
71typedef struct
72{
73 struct list_node list; /**> Intrusive doubly-circular list node */
74 uint32_t thread_id; /**> Unique thread ID assigned at creation */
75 uint32_t priority; /**> Scheduling priority (higher = more urgent) */
76 TaskHandle_t thread_handle; /**> Underlying FreeRTOS task handle */
77 void* init_parameter; /**> Opaque parameter passed to the thread entry */
78}thread_handle_t;
