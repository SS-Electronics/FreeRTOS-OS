2 * drv_handle.h — Generic driver handle definitions
3 *
4 * This file is part of FreeRTOS-OS Project.
5 *
6 * FreeRTOS-OS is free software: you can redistribute it and/or modify it under
7 * the terms of the GNU General Public License as published by the Free Software
8 * Foundation, either version 3 of the License, or (at your option) any later
9 * version.
10 *
11 * Design
12 * ──────
13 * Each peripheral class (UART, I2C, SPI, GPIO) has one generic handle struct:
14 *
15 * drv_uart_handle_t
16 * drv_iic_handle_t
17 * drv_spi_handle_t
18 * drv_gpio_handle_t
19 *
20 * Each handle embeds:
21 * • A vendor-specific hardware context (drv_hw_uart_ctx_t etc.) that is
22 * selected at compile time via CONFIG_DEVICE_VARIANT. For STM32 this
23 * wraps the HAL handle struct directly; for Infineon it holds register
24 * base addresses. The generic driver layer never touches this field
25 * directly — only the HAL implementation files do.
26 *
27 * • A const pointer to a HAL-ops table (drv_uart_hal_ops_t etc.) that is
28 * populated at registration time. All driver operations go through this
29 * table so the generic layer is vendor-agnostic.
30 *
31 * Management threads (drivers/mgmt/xxx_mgmt.c) own the handle arrays and
32 * expose typed accessor macros so upper layers never manipulate handles
33 * directly.
34 */
35

178/* ── UART ops ─────────────────────────────────────────────────────────── */
179typedef struct drv_uart_hal_ops {
180 /**
181 * @brief Configure the hardware peripheral for the settings stored in
182 * the handle's @c hw context. Called once before first use.
183 * @return OS_ERR_NONE on success, negative OS_ERR_* on failure.
184 */
185 int32_t (*hw_init)(drv_uart_handle_t *h);
186
187 /** @brief Undo hw_init — disable clock, release GPIO, etc. */
188 int32_t (*hw_deinit)(drv_uart_handle_t *h);
189
190 /**
191 * @brief Blocking transmit.
192 * @param data Pointer to TX buffer.
193 * @param len Number of bytes to send.
194 * @param timeout_ms Timeout in milliseconds.
195 */
196 int32_t (*transmit)(drv_uart_handle_t *h,
197 const uint8_t *data,
198 uint16_t len,
199 uint32_t timeout_ms);
200
201 /**
202 * @brief Blocking receive.
203 * @param data Pointer to RX buffer.
204 * @param len Number of bytes to receive.
205 * @param timeout_ms Timeout in milliseconds.
206 */
207 int32_t (*receive)(drv_uart_handle_t *h,
208 uint8_t *data,
209 uint16_t len,
210 uint32_t timeout_ms);
211
212 /**
213 * @brief Kick interrupt-driven TX from the TX ring buffer.
214 * Enables the TXE interrupt so the ISR drains the ring buffer.
215 * Safe to call from task context while TX is already in progress
216 * (re-enabling an already-set interrupt enable bit is a no-op).
217 */
218 void (*tx_start)(drv_uart_handle_t *h);
219} drv_uart_hal_ops_t;
