12 * received byte is read from DR and forwarded via drv_irq_dispatch_from_isr

38static int32_t stm32_uart_hw_init(drv_uart_handle_t *h)
39{
40 if (h == NULL || h->ops == NULL)
41 return OS_ERR_OP;
42
43 UART_HandleTypeDef *huart = &h->hw.huart;
44 const board_uart_desc_t *d = board_find_uart(huart->Instance);
45 if (d == NULL)
46 return OS_ERR_OP;
47
48 /* GPIO and NVIC setup — owned here so HAL_UART_MspInit (called internally
49 * by HAL_UART_Init below) is a no-op and init ownership stays in this fn. */
50 GPIO_InitTypeDef gpio = {
51 .Mode = d->tx_pin.mode,
52 .Pull = d->tx_pin.pull,
53 .Speed = d->tx_pin.speed,
54 .Alternate = d->tx_pin.alternate,
55 };
56
57 gpio.Pin = d->tx_pin.pin;
58 HAL_GPIO_Init(d->tx_pin.port, &gpio);
59
60 gpio.Pin = d->rx_pin.pin;
61 gpio.Alternate = d->rx_pin.alternate;
62 HAL_GPIO_Init(d->rx_pin.port, &gpio);
63
64 drv_irq_enable((int32_t)d->irqn, d->irq_priority);
65
66 huart->Init.BaudRate = h->baudrate;
67
68 HAL_StatusTypeDef ret = HAL_UART_Init(huart);
69 if (ret == HAL_OK)
70 {
71 h->initialized = 1;
72 /* Enable RXNE interrupt directly — no HAL state machine involved */
73 SET_BIT(huart->Instance->CR1, USART_CR1_RXNEIE);
74 }
75
76 h->last_err = _hal_to_os_err(ret);
77 return h->last_err;
78}

80static int32_t stm32_uart_hw_deinit(drv_uart_handle_t *h)
81{
82 if (h == NULL)
83 return OS_ERR_OP;
84
85 const board_uart_desc_t *d = board_find_uart(h->hw.huart.Instance);
86
87 CLEAR_BIT(h->hw.huart.Instance->CR1, USART_CR1_RXNEIE);
88 HAL_StatusTypeDef ret = HAL_UART_DeInit(&h->hw.huart);
89 h->initialized = 0;
90 h->last_err = _hal_to_os_err(ret);
91
92 if (d != NULL)
93 {
94 drv_irq_disable((int32_t)d->irqn);
95 HAL_GPIO_DeInit(d->tx_pin.port, d->tx_pin.pin);
96 HAL_GPIO_DeInit(d->rx_pin.port, d->rx_pin.pin);
97 }
98
99 return h->last_err;
100}

179/*
180 * hal_uart_stm32_irq_handler — direct USART register-level IRQ handler.
181 *
182 * Reads SR, CR1, CR3 and dispatches events to the drv_irq framework without
183 * going through the STM32 HAL callback chain (no HAL_UART_IRQHandler call).
184 *
185 * RX path:
186 * RXNE set → read DR (clears RXNE) → drv_irq_dispatch_from_isr(RX)
187 * subscriber (_uart_rx_cb) writes byte into IPC ring buffer
188 * Error path:
189 * Read DR to clear SR error flags, dispatch RX byte if RXNE was also set,
190 * then dispatch ERROR event.
191 * TX path:
192 * TXE/TC → clear interrupt enable bit → dispatch TX_DONE event.
193 */
194void hal_uart_stm32_irq_handler(USART_TypeDef *instance)
195{
196 for (uint8_t id = 0; id < NO_OF_UART; id++)
197 {
198 drv_uart_handle_t *h = drv_uart_get_handle(id);
199 if (h == NULL || !h->initialized || h->hw.huart.Instance != instance)
200 continue;
201
202 uint32_t sr = READ_REG(instance->SR);
203 uint32_t cr1 = READ_REG(instance->CR1);
204 uint32_t cr3 = READ_REG(instance->CR3);
205 uint32_t errorflags = sr & (USART_SR_PE | USART_SR_FE |
206 USART_SR_ORE | USART_SR_NE);
207 BaseType_t hpt = pdFALSE;
208
209 /* ── No errors: handle RX ─────────────────────────────────────────── */
210 if (errorflags == 0U)
211 {
212 if ((sr & USART_SR_RXNE) && (cr1 & USART_CR1_RXNEIE))
213 {
214 uint8_t byte = (uint8_t)(READ_REG(instance->DR) & 0xFFU);
215 drv_irq_dispatch_from_isr(IRQ_ID_UART_RX(id), &byte, &hpt);
216 portYIELD_FROM_ISR(hpt);
217 return;
218 }
219 }
220
221 /* ── Error path ───────────────────────────────────────────────────── */
222 if (errorflags && ((cr3 & USART_CR3_EIE) ||
223 (cr1 & (USART_CR1_RXNEIE | USART_CR1_PEIE))))
224 {
225 /* SR read above + DR read clears error flags on STM32F4 */
226 uint8_t byte = (uint8_t)(READ_REG(instance->DR) & 0xFFU);
227
228 if (sr & USART_SR_RXNE)
229 drv_irq_dispatch_from_isr(IRQ_ID_UART_RX(id), &byte, &hpt);
230
231 drv_irq_dispatch_from_isr(IRQ_ID_UART_ERROR(id), &errorflags, &hpt);
232 portYIELD_FROM_ISR(hpt);
233 return;
234 }
235
236 /* ── TX empty ─────────────────────────────────────────────────────── */
237 if ((sr & USART_SR_TXE) && (cr1 & USART_CR1_TXEIE))
238 {
239 uint8_t byte;
240 if (drv_uart_tx_get_next_byte(id, &byte) == OS_ERR_NONE)
241 {
242 WRITE_REG(instance->DR, byte);
243 }
244 else
245 {
246 CLEAR_BIT(instance->CR1, USART_CR1_TXEIE);
247 drv_irq_dispatch_from_isr(IRQ_ID_UART_TX_DONE(id), NULL, &hpt);
248 }
249 portYIELD_FROM_ISR(hpt);
250 return;
251 }
252
253 /* ── Transmit complete ────────────────────────────────────────────── */
254 if ((sr & USART_SR_TC) && (cr1 & USART_CR1_TCIE))
255 {
256 CLEAR_BIT(instance->CR1, USART_CR1_TCIE);
257 drv_irq_dispatch_from_isr(IRQ_ID_UART_TX_DONE(id), NULL, &hpt);
258 portYIELD_FROM_ISR(hpt);
259 return;
260 }
261
262 return;
263 }
264}
