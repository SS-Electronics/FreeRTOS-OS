42kernel_status_type drv_pca9685_iic_set_bit(uint8_t dev_address, uint8_t reg_address, uint8_t bit_pos, uint8_t val)
43{
44 kernel_status_type op_status = KERNEL_OK;
45 uint8_t reg_val = 0;
46 uint8_t data[2];
47
48 reg_val = drv_pca9685_iic_read_register(dev_address, reg_address);
49
50 reg_val &= ~((1 << REG_MODE_1_RESTART_BIT_POS) | (1 << bit_pos));
51 reg_val |= (val & 1) << bit_pos;
52
53 data[0] = reg_val;
54 op_status |= drv_pca9685_iic_write_register(dev_address, reg_address, data, 1);
55
56 return op_status;
57}

71kernel_status_type drv_pca9685_iic_set_pwm_freq(uint8_t dev_address, uint32_t frequency)
72{
73 kernel_status_type op_status;
74 float PrescalerVal;
75 uint8_t Prescale;
76
77 if (frequency >= 1526) {
78 Prescale = 0x03;
79 } else if (frequency <= 24) {
80 Prescale = 0xFF;
81 } else {
82 PrescalerVal = (25000000 / (4096.0 * (float) frequency)) - 1;
83 Prescale = floor(PrescalerVal + 0.5);
84 }
85 op_status = KERNEL_OK;
86 /* to set the PWN frequencty the controller has to be in sleep mode */
87
88 /* write the PWM prescaler */
89 data_buffer[0] = Prescale;
90 op_status |= drv_pca9685_iic_write_register(dev_address, REG_PRESCALE, data_buffer, 1);
91 drv_pca9685_iic_set_bit(dev_address, REG_MODE_0, REG_MODE_1_SLEEP_BIT_POS, 1);
92 drv_time_delay_ms(100);
93 /* Starting normal operation */
94 drv_pca9685_iic_set_bit(dev_address, REG_MODE_0, REG_MODE_1_SLEEP_BIT_POS, 0);
95 drv_time_delay_ms(100);
96 drv_pca9685_iic_set_bit(dev_address, REG_MODE_0, REG_MODE_1_RESTART_BIT_POS, 1);
97 drv_time_delay_ms(100);
98
99 return op_status;
100}

102kernel_status_type drv_pca9685_iic_set_pwm(uint8_t dev_address, uint8_t channel, uint16_t on_time, uint16_t off_time)
103{
104 kernel_status_type op_status = KERNEL_OK;
105 uint8_t reg_address;
106 uint8_t data[4];
107
108 reg_address = REG_LED0_ON_L + (4 * channel);
109
110 data[0] = on_time & 0xFF;
111 data[1] = (on_time >> 8) & 0x0F;
112 data[2] = off_time & 0xFF;
113 data[3] = (off_time >> 8) & 0x0F;
114
115 op_status |= drv_pca9685_iic_write_register(dev_address, reg_address, data, 4);
116
117 return op_status;
118}

148kernel_status_type drv_pca9685_iic_write_register(uint8_t device_address, uint8_t reg_address, uint8_t* data , uint8_t length)
149{
150 if(length == 1) // single register write
151 {
152 iic_pdu_buffer.device_id = PCA9685_COMM;
153 iic_pdu_buffer.iic_operation = IIC_OPERATION_REG_WRITE;
154 iic_pdu_buffer.slave_address = device_address;
155 iic_pdu_buffer.reg_addr = reg_address;
156 iic_pdu_buffer.txn_data[0] = data[0];
157 /* do a IIC operation */
158 xQueueSend( pipe_iic_pdu_tx_handle,
159 &iic_pdu_buffer,
160 (TickType_t)0
161 );
162 }
163 else
164 {
165 iic_pdu_buffer.device_id = PCA9685_COMM;
166 iic_pdu_buffer.iic_operation = IIC_OPERATION_WRITE;
167 iic_pdu_buffer.slave_address = device_address;
168 iic_pdu_buffer.txn_data[0] = reg_address;
169 iic_pdu_buffer.length = length + 1;
170 for(int i = 0; i < length; i++)
171 {
172 iic_pdu_buffer.txn_data[i+1] = data[i];
173 }
174 /* do a IIC operation */
175 xQueueSend( pipe_iic_pdu_tx_handle,
176 &iic_pdu_buffer,
177 (TickType_t)0
178 );
179 }
180 /* Mark the current time */
181 iic_pdu_buffer.op_status = KERNEL_ERR;
182 set_time_ms = drv_time_get_ticks();
183 /* wait for the feedback */
184 while((drv_time_get_ticks() - set_time_ms) < DRV_PCA9685_ACK_TIMEOUT_MS)
185 {
186 xQueuePeek( pipe_iic_pdu_rx_handle,
187 &iic_pdu_buffer,
188 (TickType_t)0
189 );
190 /* If the PDU receive for */
191 if((iic_pdu_buffer.device_id == PCA9685_COMM) && \
192 (iic_pdu_buffer.slave_address == device_address))
193 {
194 xQueueReceive( pipe_iic_pdu_rx_handle,
195 &iic_pdu_buffer,
196 (TickType_t)0
197 );
198 break;
199 }
200 }
201
202 return iic_pdu_buffer.op_status;
203}
