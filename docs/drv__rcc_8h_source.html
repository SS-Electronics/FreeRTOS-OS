1/*
2 * drv_rcc.h — Reset and Clock Control driver public API
3 *
4 * This file is part of FreeRTOS-OS Project.
5 *
6 * Portable interface to the clock initialisation subsystem. The vendor-
7 * specific HAL backend (hal_rcc_stm32.c, etc.) is selected at compile time
8 * via CONFIG_DEVICE_VARIANT and registers its ops table inside
9 * drv_rcc_clock_init(). Callers never need to include a vendor header.
10 *
11 * Call sequence (from main, before the scheduler):
12 * HAL_Init();
13 * drv_rcc_clock_init(); ← configures PLL, flash latency, bus dividers
14 */
15
