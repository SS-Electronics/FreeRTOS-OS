1/*
2 * irq_notify.h — Generic vendor-agnostic IRQ notification framework
3 *
4 * This file is part of FreeRTOS-OS Project.
5 *
6 * Design
7 * ──────
8 * Hardware IRQ events are mapped to generic numeric IRQ IDs defined by this
9 * header. Any driver or service can subscribe to an IRQ ID via irq_register().
10 * When the hardware fires, the vendor HAL callback calls irq_notify_from_isr()
11 * which dispatches to every registered callback in ISR context.
12 *
13 * Naming convention
14 * ─────────────────
15 * IRQ_ID_UART_RX(n) — byte received on UART bus n
16 * IRQ_ID_UART_TX_DONE(n) — transmit complete on UART bus n
17 * IRQ_ID_UART_ERROR(n) — UART error on bus n
18 * IRQ_ID_I2C_TX_DONE(n) — I2C master transmit complete on bus n
19 * IRQ_ID_I2C_RX_DONE(n) — I2C master receive complete on bus n
20 * IRQ_ID_I2C_ERROR(n) — I2C error on bus n
21 * IRQ_ID_SPI_TX_DONE(n) — SPI transmit complete on bus n
22 * IRQ_ID_SPI_RX_DONE(n) — SPI receive complete on bus n
23 * IRQ_ID_SPI_TXRX_DONE(n) — SPI full-duplex transfer complete on bus n
24 * IRQ_ID_SPI_ERROR(n) — SPI error on bus n
25 */

64/**
