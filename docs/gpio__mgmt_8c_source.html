19static void gpio_mgmt_thread(void *arg)
20{
21 (void)arg;
22
23 os_thread_delay(TIME_OFFSET_GPIO_MANAGEMENT);
24
25 /* Register all GPIO lines described in the board configuration.
26 * For each line: enable the GPIO clock, configure the pin parameters
27 * in the hw context, then call drv_gpio_register() which calls hw_init(). */
28 {
29 const drv_gpio_hal_ops_t *ops =
30#if (CONFIG_DEVICE_VARIANT == MCU_VAR_STM)
31 hal_gpio_stm32_get_ops();
32#else
33 NULL;
34#endif
35 if (ops != NULL)
36 {
37 const board_config_t *bc = board_get_config();
38 for (uint8_t i = 0; i < bc->gpio_count; i++)
39 {
40 const board_gpio_desc_t *d = &bc->gpio_table[i];
41 drv_gpio_handle_t *h = drv_gpio_get_handle(d->dev_id);
42
43#if (CONFIG_DEVICE_VARIANT == MCU_VAR_STM)
44 board_clk_enable();
45 hal_gpio_stm32_set_config(h,
46 d->port,
47 d->pin,
48 d->mode,
49 d->pull,
50 d->speed,
51 d->active_state);
52#endif
53 int32_t err = drv_gpio_register(d->dev_id, ops);
54 (void)err;
55
56 /* Apply initial state for output pins */
57 if (h->initialized && d->initial_state && h->ops != NULL)
58 h->ops->set(h);
59 }
60 }
61 }
62
63 gpio_mgmt_msg_t msg;
64
65 for (;;)
66 {
67 if (xQueueReceive(_mgmt_queue, &msg, portMAX_DELAY) != pdTRUE)
68 continue;
69
70 /* Apply optional delay before executing */
71 if (msg.delay_ms > 0)
72 os_thread_delay(msg.delay_ms);
73
74 drv_gpio_handle_t *h = drv_gpio_get_handle(msg.gpio_id);
75 if (h == NULL || h->ops == NULL || !h->initialized)
76 continue;
77
78 switch (msg.cmd)
79 {
80 case GPIO_MGMT_CMD_SET: h->ops->set(h); break;
81 case GPIO_MGMT_CMD_CLEAR: h->ops->clear(h); break;
82 case GPIO_MGMT_CMD_TOGGLE: h->ops->toggle(h); break;
83 case GPIO_MGMT_CMD_WRITE: h->ops->write(h, msg.state); break;
84 case GPIO_MGMT_CMD_REINIT:
85 h->ops->hw_deinit(h);
86 h->ops->hw_init(h);
87 break;
88 default: break;
89 }
90 }
91}

93int32_t gpio_mgmt_start(void)
94{
95 _mgmt_queue = xQueueCreate(GPIO_MGMT_QUEUE_DEPTH, sizeof(gpio_mgmt_msg_t));
96 if (_mgmt_queue == NULL)
97 return OS_ERR_MEM_OF;
98
99 int32_t tid = os_thread_create(gpio_mgmt_thread,
100 "gpio_mgmt",
101 PROC_SERVICE_GPIO_MGMT_STACK_SIZE,
102 PROC_SERVICE_GPIO_MGMT_PRIORITY,
103 NULL);
104 return (tid >= 0) ? OS_ERR_NONE : OS_ERR_OP;
105}

112int32_t gpio_mgmt_post(uint8_t gpio_id, gpio_mgmt_cmd_t cmd,
113 uint8_t state, uint32_t delay_ms)
114{
115 if (_mgmt_queue == NULL)
116 return OS_ERR_OP;
117
118 gpio_mgmt_msg_t msg = {
119 .cmd = cmd,
120 .gpio_id = gpio_id,
121 .state = state,
122 .delay_ms = delay_ms,
123 };
124
125 return (xQueueSend(_mgmt_queue, &msg, 0) == pdTRUE) ? OS_ERR_NONE : OS_ERR_OP;
