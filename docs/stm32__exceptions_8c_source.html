1/**
2 * @file stm32_exceptions.c
3 * @author Subhajit Roy (subhajitroy005@gmail.com)
4 * @brief Cortex-M4 core exception handlers for STM32
5 *
6 * @details
7 * Strong definitions for Cortex-M core exception handlers that override the
8 * weak symbols in startup_stm32f411vetx.c.
9 *
10 * Fault handler chain (HardFault / MemManage / BusFault / UsageFault):
11 *
12 * CPU fires exception
13 * │
14 * ▼ naked trampoline (_Handler)
15 * determine active SP from EXC_RETURN bit[2]:
16 * bit[2]==0 → MSP was active (running in handler or MSP task)
17 * bit[2]==1 → PSP was active (running in a FreeRTOS task)
18 * │
19 * ▼ _fault_handler_c(frame*, exc_return, fault_type)
20 * capture hardware exception frame (r0–r3, r12, lr, pc, xpsr)
21 * capture SCB->HFSR, CFSR, BFAR, MMFAR
22 * write g_fault_info (visible in debugger / crash dump)
23 * emit diagnostic text directly via USART2 (no ring buffer, no RTOS)
24 * halt in spin loop
25 *
26 * @note
27 * This module is compiled only when CONFIG_DEVICE_VARIANT == MCU_VAR_STM.
28 *
29 * This file is part of FreeRTOS-OS Project.
30 */

73/* ── Direct USART2 output — safe in fault context ───────────────────────── */
74/*
75 * USART2 APB1 base: 0x40004400
76 * SR offset 0x00 — bit 7 TXE, bit 6 TC
77 * DR offset 0x04
78 */
79
80#define USART2_BASE 0x40004400UL
81#define USART2_SR (*(volatile uint32_t *)(USART2_BASE + 0x00U))
82#define USART2_DR (*(volatile uint32_t *)(USART2_BASE + 0x04U))
83#define USART2_TXE (1UL << 7)
84#define USART2_TC (1UL << 6)
85
86static void _fault_putchar(char c)
87{
88 while (!(USART2_SR & USART2_TXE))
89 ;
90 USART2_DR = (uint32_t)(uint8_t)c;
91}

93static void _fault_puts(const char *s)
94{
95 while (*s)
96 _fault_putchar(*s++);
97}

99static void _fault_print_hex32(uint32_t v)
100{
101 static const char hex[] = "0123456789ABCDEF";
102 _fault_putchar('0');
103 _fault_putchar('x');
104 for (int i = 28; i >= 0; i -= 4)
105 _fault_putchar(hex[(v >> i) & 0xFU]);
106}

108static void _fault_print_dec(uint32_t v)
109{
110 if (v == 0) { _fault_putchar('0'); return; }
111 char buf[10];
112 int n = 0;
113 while (v) { buf[n++] = (char)('0' + v % 10U); v /= 10U; }
114 while (n > 0) { n--; _fault_putchar(buf[n]); }
115}

117/* ── CFSR / HFSR bit decoders ───────────────────────────────────────────── */
118
119static void _decode_mmfsr(uint32_t cfsr)
120{
121 uint32_t m = cfsr & 0xFFU;
122 if (!m) return;
123 _fault_puts(" MMFSR:");
124 if (m & (1U << 7)) _fault_puts(" MMARVALID");
125 if (m & (1U << 5)) _fault_puts(" MLSPERR");
126 if (m & (1U << 4)) _fault_puts(" MSTKERR");
127 if (m & (1U << 3)) _fault_puts(" MUNSTKERR");
128 if (m & (1U << 1)) _fault_puts(" DACCVIOL");
129 if (m & (1U << 0)) _fault_puts(" IACCVIOL");
130 _fault_puts("\r\n");
131}
132
133static void _decode_bfsr(uint32_t cfsr)
134{
135 uint32_t b = (cfsr >> 8U) & 0xFFU;
136 if (!b) return;
137 _fault_puts(" BFSR:");
138 if (b & (1U << 7)) _fault_puts(" BFARVALID");
139 if (b & (1U << 5)) _fault_puts(" LSPERR");
140 if (b & (1U << 4)) _fault_puts(" STKERR");
141 if (b & (1U << 3)) _fault_puts(" UNSTKERR");
142 if (b & (1U << 2)) _fault_puts(" IMPRECISERR");
143 if (b & (1U << 1)) _fault_puts(" PRECISERR");
144 if (b & (1U << 0)) _fault_puts(" IBUSERR");
145 _fault_puts("\r\n");
146}
147
148static void _decode_ufsr(uint32_t cfsr)
149{
150 uint32_t u = (cfsr >> 16U) & 0xFFFFU;
151 if (!u) return;
152 _fault_puts(" UFSR:");
153 if (u & (1U << 9)) _fault_puts(" DIVBYZERO");
154 if (u & (1U << 8)) _fault_puts(" UNALIGNED");
155 if (u & (1U << 3)) _fault_puts(" NOCP");
156 if (u & (1U << 2)) _fault_puts(" INVPC");
157 if (u & (1U << 1)) _fault_puts(" INVSTATE");
158 if (u & (1U << 0)) _fault_puts(" UNDEFINSTR");
159 _fault_puts("\r\n");
160}

162static void _decode_hfsr(uint32_t hfsr)
163{
164 _fault_puts(" HFSR:");
165 if (hfsr & (1U << 31)) _fault_puts(" DEBUGEVT");
166 if (hfsr & (1U << 30)) _fault_puts(" FORCED");
167 if (hfsr & (1U << 1)) _fault_puts(" VECTBL");
168 _fault_puts("\r\n");
169}

171/* ── Central fault C handler ─────────────────────────────────────────────── */

173static const char *_fault_names[] = {
174 "HardFault", "MemManage", "BusFault", "UsageFault"
175};
176
177void __attribute__((used, noinline)) _fault_handler_c(uint32_t *frame, uint32_t exc_return, uint32_t fault_type)
178{
179 /* Capture SCB fault registers before anything can modify them */
180 uint32_t hfsr = *(volatile uint32_t *)0xE000ED2CUL;
181 uint32_t cfsr = *(volatile uint32_t *)0xE000ED28UL;
182 uint32_t mmfar = *(volatile uint32_t *)0xE000ED34UL;
183 uint32_t bfar = *(volatile uint32_t *)0xE000ED38UL;
184
185 /* Fill diagnostic record */
186 g_fault_info.frame.r0 = frame[0];
187 g_fault_info.frame.r1 = frame[1];
188 g_fault_info.frame.r2 = frame[2];
189 g_fault_info.frame.r3 = frame[3];
190 g_fault_info.frame.r12 = frame[4];
191 g_fault_info.frame.lr = frame[5];
192 g_fault_info.frame.pc = frame[6];
193 g_fault_info.frame.xpsr = frame[7];
194 g_fault_info.hfsr = hfsr;
195 g_fault_info.cfsr = cfsr;
196 g_fault_info.bfar = bfar;
197 g_fault_info.mmfar = mmfar;
198 g_fault_info.exc_return = exc_return;
199 g_fault_info.fault_type = fault_type;
200
201 /* Emit diagnostic over USART2 */
202 _fault_puts("\r\n\r\n*** ");
203 _fault_puts(fault_type < 4U ? _fault_names[fault_type] : "Fault");
204 _fault_puts(" ***\r\n");
205
206 _fault_puts("PC : "); _fault_print_hex32(frame[6]); _fault_puts("\r\n");
207 _fault_puts("LR : "); _fault_print_hex32(frame[5]); _fault_puts("\r\n");
208 _fault_puts("SP : "); _fault_print_hex32((uint32_t)frame); _fault_puts("\r\n");
209 _fault_puts("R0 : "); _fault_print_hex32(frame[0]); _fault_puts("\r\n");
210 _fault_puts("R1 : "); _fault_print_hex32(frame[1]); _fault_puts("\r\n");
211 _fault_puts("R2 : "); _fault_print_hex32(frame[2]); _fault_puts("\r\n");
212 _fault_puts("R3 : "); _fault_print_hex32(frame[3]); _fault_puts("\r\n");
213 _fault_puts("R12 : "); _fault_print_hex32(frame[4]); _fault_puts("\r\n");
214 _fault_puts("xPSR : "); _fault_print_hex32(frame[7]); _fault_puts("\r\n");
215
216 _fault_puts("HFSR : "); _fault_print_hex32(hfsr); _fault_puts("\r\n");
217 _fault_puts("CFSR : "); _fault_print_hex32(cfsr); _fault_puts("\r\n");
218
219 _decode_hfsr(hfsr);
220 _decode_mmfsr(cfsr);
221 _decode_bfsr(cfsr);
222 _decode_ufsr(cfsr);
223
224 if ((cfsr >> 8U) & (1U << 7)) /* BFARVALID */
225 {
226 _fault_puts("BFAR : "); _fault_print_hex32(bfar); _fault_puts("\r\n");
227 }
228 if (cfsr & (1U << 7)) /* MMARVALID */
229 {
230 _fault_puts("MMFAR: "); _fault_print_hex32(mmfar); _fault_puts("\r\n");
231 }
232
233 _fault_puts("EXC_RETURN: "); _fault_print_hex32(exc_return);
234 _fault_puts((exc_return & 4U) ? " (PSP)\r\n" : " (MSP)\r\n");
235
236 /* Wait for last byte to drain */
237 while (!(USART2_SR & USART2_TC))
238 ;
239
240 for (;;)
241 ; /* halt — attach debugger, inspect g_fault_info */
242}

255/* ── Fault trampolines ───────────────────────────────────────────────────── */
256/*
257 * Each naked trampoline:
258 * 1. Tests EXC_RETURN bit[2] to pick MSP or PSP as the exception frame ptr.
259 * 2. Passes it as r0, EXC_RETURN as r1, fault type constant as r2.
260 * 3. Branches (not calls) to _fault_handler_c — it never returns.
261 */
262
263__attribute__((naked)) void HardFault_Handler(void)
264{
265 __asm volatile(
266 "tst lr, #4 \n"
267 "ite eq \n"
268 "mrseq r0, msp \n"
269 "mrsne r0, psp \n"
270 "mov r1, lr \n"
271 "mov r2, %0 \n"
272 "b _fault_handler_c\n"
273 :
274 : "i" (FAULT_TYPE_HARD)
275 );
276}
