1/*
2 * drv_dma.h — Linux-style DMA engine API
3 *
4 * This file is part of FreeRTOS-OS Project.
5 *
6 * Architecture mirrors Linux dmaengine:
7 *
8 * dma_device — registered DMA controller (DMA1 / DMA2)
9 * dma_chan — one stream/channel inside a controller
10 * dma_async_tx_descriptor — describes a single transfer
11 * dma_slave_config — peripheral-side parameters
12 *
13 * Typical usage (peripheral slave transfer):
14 *
15 * // 1. Request a channel once at init
16 * dma_chan_t *ch = dma_request_chan("DMA1", 3);
17 *
18 * // 2. Configure slave (peripheral address, direction, widths)
19 * dma_slave_config_t cfg = {
20 * .direction = DMA_MEM_TO_DEV,
21 * .dst_addr = (uint32_t)&USART2->DR,
22 * .src_addr_width = DMA_SLAVE_BUSWIDTH_1_BYTE,
23 * .dst_addr_width = DMA_SLAVE_BUSWIDTH_1_BYTE,
24 * .src_addr_adj = DMA_ADDR_INCREMENT,
25 * .dst_addr_adj = DMA_ADDR_FIXED,
26 * .request = 4, // DMA channel 4 → USART2_TX on DMA1 Stream6
27 * };
28 * dmaengine_slave_config(ch, &cfg);
29 *
30 * // 3. Prepare descriptor
31 * dma_async_tx_descriptor_t *desc =
32 * dmaengine_prep_slave_single(ch, (uint32_t)buf, len, DMA_MEM_TO_DEV);
33 * desc->callback = my_tx_done;
34 * desc->callback_param = my_ctx;
35 *
36 * // 4. Submit and start
37 * dmaengine_submit(desc);
38 * dma_async_issue_pending(ch);
39 *
40 * // 5. Release when done
41 * dma_release_chan(ch);
42 */
43

118/* ── Slave configuration ──────────────────────────────────────────────────── */
119/*
120 * For MEM_TO_DEV (e.g. UART TX):
121 * src_addr_adj = DMA_ADDR_INCREMENT (walk memory buffer)
122 * dst_addr_adj = DMA_ADDR_FIXED (peripheral DR stays put)
123 * dst_addr = &USARTx->DR
124 *
125 * For DEV_TO_MEM (e.g. UART RX):
126 * src_addr_adj = DMA_ADDR_FIXED
127 * dst_addr_adj = DMA_ADDR_INCREMENT
128 * src_addr = &USARTx->DR
129 *
130 * request: HW request line / channel mux index (0–7 for STM32F4)
131 */
132typedef struct {
133 dma_transfer_direction_t direction;
134 uint32_t src_addr; /* peripheral register address */
135 uint32_t dst_addr; /* peripheral register address */
136 dma_slave_buswidth_t src_addr_width;
137 dma_slave_buswidth_t dst_addr_width;
138 uint32_t src_maxburst; /* beats per burst (1 = no burst) */
139 uint32_t dst_maxburst;
140 dma_addr_adj_t src_addr_adj;
141 dma_addr_adj_t dst_addr_adj;
142 uint8_t request; /* channel mux (DMA_CHANNEL_x) */
143} dma_slave_config_t;
144

164} dma_async_tx_descriptor_t;
