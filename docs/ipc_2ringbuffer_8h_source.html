3 *
4 * This file is part of FreeRTOS-OS Project.
5 *
6 * Dynamic model:
7 * ringbuffer_create(size) — allocates descriptor + data buffer via kmaloc()
8 * ringbuffer_destroy(rb) — frees both via kfree()

30/*
31 * Mirror-bit ring buffer.
32 *
33 * read_index / write_index are 15-bit values (max buffer size 32 KiB).
34 * The single mirror bit distinguishes full from empty when both indices
35 * are equal: same mirror → empty, different mirror → full.
36 */
37struct ringbuffer {
38 uint8_t *buffer_ptr;
39 uint16_t read_mirror : 1;
40 uint16_t read_index : 15;
41 uint16_t write_mirror : 1;
42 uint16_t write_index : 15;
43 int16_t buffer_size;
44};
45
46enum ringbuffer_state {
47 RT_RINGBUFFER_EMPTY,
48 RT_RINGBUFFER_FULL,
49 RT_RINGBUFFER_HALFFULL,
50};
