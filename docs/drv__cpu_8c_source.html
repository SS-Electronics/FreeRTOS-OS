36static drv_wdg_handle_t _wdg_handle;
37
38int32_t drv_wdg_register(const drv_wdg_hal_ops_t *ops)
39{
40 if (ops == NULL)
41 return OS_ERR_OP;
42
43 _wdg_handle.ops = ops;
44 _wdg_handle.initialized = 0;
45 _wdg_handle.last_err = OS_ERR_NONE;
46
47 int32_t err = _wdg_handle.ops->hw_init(&_wdg_handle);
48 if (err != OS_ERR_NONE)
49 return err;
50
51 _wdg_handle.initialized = 1;
52 return OS_ERR_NONE;
53}
54
55drv_wdg_handle_t *drv_wdg_get_handle(void)
56{
57 return &_wdg_handle;
58}
59
60void drv_wdg_refresh(void)
61{
62 if (!_wdg_handle.initialized || _wdg_handle.ops == NULL)
63 return;
64 _wdg_handle.ops->refresh(&_wdg_handle);
65}

67#else
