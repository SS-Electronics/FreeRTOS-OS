49/**
50 * @brief A generic thread create api.
51 * @note Creates a dynamic FreeRTOS task and inserts the new
52 * thread_handle_t into the intrusive doubly-circular thread_list
53 * using list_add_tail(), mirroring the Linux kernel task list.
54 * @param thread_func Thread entry function pointer.
55 * @param thread_name Name string for debugging.
56 * @param thread_stack_depth Stack size in words.
57 * @param priority Scheduling priority (0 = idle).
58 * @param thread_parameter Opaque pointer forwarded to the thread.
59 * @return Positive thread_id on success, negative OS_ERR_* on failure.
60 */
61int32_t __SECTION_OS os_thread_create(thread_func_t thread_func,
62 const char * const thread_name,
63 uint32_t thread_stack_depth,
64 uint32_t priority,
65 void * thread_parameter)
66{
67 if( (thread_func != NULL) && (thread_stack_depth != 0) )
68 {
69 thread_handle_t* handle = (thread_handle_t *) kmaloc(sizeof(thread_handle_t));
70
71 if(handle != NULL)
72 {
73 handle->init_parameter = thread_parameter;
74 handle->priority = priority;
75
76 BaseType_t returned = xTaskCreate(thread_func,
77 thread_name,
78 thread_stack_depth,
79 thread_parameter,
80 (UBaseType_t)priority,
81 &handle->thread_handle);
82
83 if(returned == pdTRUE)
84 {
85 thread_counter++;
86 handle->thread_id = thread_counter;
87 }
88 else
89 {
90 kfree(handle);
91 return OS_ERR_OP;
92 }
93
94 /* Append to the tail of the thread list (Linux task_list style) */
95 list_add_tail(&handle->list, &thread_list);
96
97 return handle->thread_id;
98 }
99 else
100 {
101 return OS_ERR_MEM_OP;
102 }
103 }
104 else
105 {
106 return OS_ERR_OP;
107 }
108}

111/**
112 * @brief Delete a thread by ID and free its resources.
113 * @note Walks the intrusive list with list_for_each_entry(),
114 * calls vTaskDelete(), unlinks via list_delete(), then frees.
115 * @param thread_id ID returned by os_thread_create().
116 * @return OS_ERR_NONE on success, OS_ERR_OP if the ID is not found.
117 */
118int32_t __SECTION_OS os_thread_delete(uint32_t thread_id)
119{
120 if(thread_id == 0)
121 return OS_ERR_OP;
122
123 thread_handle_t *pos;
124
125 list_for_each_entry(pos, &thread_list, list)
126 {
127 if(pos->thread_id == thread_id)
128 {
129 vTaskDelete(pos->thread_handle);
130 list_delete(&pos->list);
131 kfree(pos);
132 return OS_ERR_NONE;
133 }
134 }
135
136 return OS_ERR_OP;
137}

160/**
161 * @brief Suspend an arbitrary thread by its ID.
162 * @note Walks the intrusive thread list to locate the handle then
163 * calls vTaskSuspend(), mirroring how the Linux kernel
164 * iterates task_struct via the tasks list.
165 * @param thread_id ID returned by os_thread_create().
166 */
167void __SECTION_OS os_suspend_thread(uint32_t thread_id)
168{
169 if(thread_id == 0)
170 return;
171
172 thread_handle_t *pos;
173
174 list_for_each_entry(pos, &thread_list, list)
175 {
176 if(pos->thread_id == thread_id)
177 {
178 vTaskSuspend(pos->thread_handle);
179 return;
180 }
181 }
182}
