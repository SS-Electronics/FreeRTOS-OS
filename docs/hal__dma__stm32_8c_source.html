1/*
2 * hal_dma_stm32.c — STM32F4 DMA engine HAL backend
3 *
4 * This file is part of FreeRTOS-OS Project.
5 *
6 * Maps the dma_chan_hal_ops_t vtable onto STM32 HAL DMA primitives.
7 * Each dma_chan_t.hw_ctx points to a drv_hw_dma_chan_ctx_t that wraps a
8 * DMA_HandleTypeDef plus the stream IRQ number.
9 *
10 * ISR routing
11 * ───────────
12 * The 16 weak DMAx_StreamY_IRQHandler symbols from startup_stm32f411vetx.s
13 * are overridden here. Each handler just calls HAL_DMA_IRQHandler() which
14 * then fires _hal_cplt_cb / _hal_error_cb → dma_complete_callback /
15 * dma_error_callback in the engine core.
16 *
17 * Note on DMA_HandleTypeDef.Parent
18 * ─────────────────────────────────
19 * We store the dma_chan_t * in hdma.Parent. This field is normally used by
20 * the STM32 UART/SPI HAL to back-link to the peripheral handle, so this DMA
21 * framework must NOT be mixed with HAL_UART_Transmit_DMA() / HAL_SPI_..._DMA().
22 * Use either the framework or the HAL peripheral DMA path — not both on the
23 * same stream.
24 */
25

73static uint32_t _hal_dir(dma_transfer_direction_t d)
74{
75 switch (d) {
76 case DMA_MEM_TO_DEV: return DMA_MEMORY_TO_PERIPH;
77 case DMA_DEV_TO_MEM: return DMA_PERIPH_TO_MEMORY;
78 case DMA_MEM_TO_MEM: return DMA_MEMORY_TO_MEMORY;
79 default: return DMA_PERIPH_TO_MEMORY;
80 }
81}

114static int32_t _slave_config(dma_chan_t *chan, const dma_slave_config_t *cfg)
115{
116 drv_hw_dma_chan_ctx_t *ctx = (drv_hw_dma_chan_ctx_t *)chan->hw_ctx;
117 DMA_HandleTypeDef *h = &ctx->hdma;
118
119 h->Init.Direction = _hal_dir(cfg->direction);
120 h->Init.Channel = (uint32_t)cfg->request << DMA_SxCR_CHSEL_Pos;
121 h->Init.PeriphInc = (cfg->src_addr_adj == DMA_ADDR_INCREMENT)
122 ? DMA_PINC_ENABLE : DMA_PINC_DISABLE;
123 h->Init.MemInc = (cfg->dst_addr_adj == DMA_ADDR_INCREMENT)
124 ? DMA_MINC_ENABLE : DMA_MINC_DISABLE;
125 h->Init.PeriphDataAlignment = _hal_align(cfg->src_addr_width);
126 h->Init.MemDataAlignment = _hal_align(cfg->dst_addr_width);
127 h->Init.Mode = DMA_NORMAL;
128 h->Init.Priority = DMA_PRIORITY_HIGH;
129 h->Init.FIFOMode = DMA_FIFOMODE_DISABLE;
130 h->Init.FIFOThreshold = DMA_FIFO_THRESHOLD_FULL;
131 h->Init.MemBurst = DMA_MBURST_SINGLE;
132 h->Init.PeriphBurst = DMA_PBURST_SINGLE;
133
134 HAL_DMA_DeInit(h);
135 if (HAL_DMA_Init(h) != HAL_OK)
136 return OS_ERR_OP;
137
138 h->XferCpltCallback = _hal_cplt_cb;
139 h->XferErrorCallback = _hal_error_cb;
140 h->Parent = (void *)chan; /* back-link for ISR dispatch */
141
142 HAL_NVIC_SetPriority(ctx->irqn, 5, 0);
143 HAL_NVIC_EnableIRQ(ctx->irqn);
144
145 return OS_ERR_NONE;
146}

148static int32_t _dma_start(dma_chan_t *chan, dma_async_tx_descriptor_t *desc)
149{
150 drv_hw_dma_chan_ctx_t *ctx = (drv_hw_dma_chan_ctx_t *)chan->hw_ctx;
151 DMA_HandleTypeDef *h = &ctx->hdma;
152
153 /* Switch mode if needed */
154 uint32_t mode = (desc->mode == DMA_XFER_CYCLIC) ? DMA_CIRCULAR : DMA_NORMAL;
155 if (h->Init.Mode != mode)
156 {
157 h->Init.Mode = mode;
158 HAL_DMA_Init(h);
159 h->XferCpltCallback = _hal_cplt_cb;
160 h->XferErrorCallback = _hal_error_cb;
161 h->Parent = (void *)chan;
162 }
163
164 /* Compute element count from byte length and peripheral width */
165 const dma_slave_config_t *cfg = &chan->slave_cfg;
166 uint8_t width = (uint8_t)(cfg->src_addr_width
167 ? cfg->src_addr_width : 1U);
168 uint32_t count = desc->len / width;
169
170 if (desc->direction == DMA_MEM_TO_MEM)
171 {
172 /* memcpy: both sides increment with byte width */
173 h->Init.PeriphInc = DMA_PINC_ENABLE;
174 h->Init.MemInc = DMA_MINC_ENABLE;
175 h->Init.PeriphDataAlignment = DMA_PDATAALIGN_BYTE;
176 h->Init.MemDataAlignment = DMA_MDATAALIGN_BYTE;
177 HAL_DMA_Init(h);
178 h->XferCpltCallback = _hal_cplt_cb;
179 h->XferErrorCallback = _hal_error_cb;
180 h->Parent = (void *)chan;
181 count = desc->len;
182 }
183
184 if (HAL_DMA_Start_IT(h, desc->src, desc->dst, count) != HAL_OK)
185 return OS_ERR_OP;
186
187 return OS_ERR_NONE;
188}

246/* ── hal_dma_stm32_init ──────────────────────────────────────────────────── */

248void hal_dma_stm32_init(void)
249{
250 __HAL_RCC_DMA1_CLK_ENABLE();
251 __HAL_RCC_DMA2_CLK_ENABLE();
252
253 for (uint8_t i = 0; i < 8; i++)
254 {
255 _dma1_ctx[i].hdma.Instance = _dma1_stream[i];
256 _dma1_ctx[i].irqn = _dma1_irqn[i];
257 hal_dma1_device.channels[i].hw_ctx = &_dma1_ctx[i];
258
259 _dma2_ctx[i].hdma.Instance = _dma2_stream[i];
260 _dma2_ctx[i].irqn = _dma2_irqn[i];
261 hal_dma2_device.channels[i].hw_ctx = &_dma2_ctx[i];
262 }
263
264 dma_register_device(&hal_dma1_device);
265 dma_register_device(&hal_dma2_device);
266}

268/*
269 * hal_dma_stm32_irq_dispatch — call from the IRQ dispatch generated file.
270 *
271 * Add a <dma> entry to your irq_table.xml for each stream in use, e.g.:
272 *
273 * <dma controller="1" stream="6" channel="4" priority="5"
274 * dispatch="hal_dma_stm32_irq_dispatch(1, 6)"
275 * dispatch_include="&lt;drivers/dma/hal/stm32/hal_dma_stm32.h&gt;"/>
276 *
277 * Parameters:
278 * ctrl_idx — 1 for DMA1, 2 for DMA2
279 * stream_idx — stream number 0–7
280 */
281void hal_dma_stm32_irq_dispatch(uint8_t ctrl_idx, uint8_t stream_idx)
282{
283 if (stream_idx >= 8)
284 return;
285
286 if (ctrl_idx == 1)
287 HAL_DMA_IRQHandler(&_dma1_ctx[stream_idx].hdma);
288 else if (ctrl_idx == 2)
289 HAL_DMA_IRQHandler(&_dma2_ctx[stream_idx].hdma);
290}
