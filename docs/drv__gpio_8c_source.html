1/*
2 * drv_gpio.c — Generic GPIO driver
3 *
4 * This file is part of FreeRTOS-OS Project.
5 *
6 * Vendor-agnostic GPIO driver. All pin operations go through the
7 * drv_gpio_hal_ops_t table bound at registration time.
8 *
9 * GPIO lines are registered individually by board-specific init code
10 * (e.g. board_init() or stm32f4xx_hal_msp.c). Each logical line gets a
11 * unique dev_id assigned by the caller.
12 */
13
14#include <drivers/drv_handle.h>
15#include <board/mcu_config.h>
16#include <def_err.h>

18/* Maximum logical GPIO lines managed by this driver */
19#ifndef DRV_GPIO_MAX_LINES
20#define DRV_GPIO_MAX_LINES 32
21#endif
22
23/* ── Handle storage ───────────────────────────────────────────────────── */
24
25static drv_gpio_handle_t _gpio_handles[DRV_GPIO_MAX_LINES];
26static uint8_t _gpio_registered_count = 0;

28/* ── Registration API ─────────────────────────────────────────────────── */
29
30int32_t drv_gpio_register(uint8_t dev_id, const drv_gpio_hal_ops_t *ops)
31{
32 if (dev_id >= DRV_GPIO_MAX_LINES || ops == NULL)
33 return OS_ERR_OP;
34
35 drv_gpio_handle_t *h = &_gpio_handles[dev_id];
36
37 h->dev_id = dev_id;
38 h->ops = ops;
39 h->initialized = 0;
40 h->last_err = OS_ERR_NONE;
41
42 if (_gpio_registered_count <= dev_id)
43 _gpio_registered_count = (uint8_t)(dev_id + 1);
44
45 return h->ops->hw_init(h);
46}
47
48/* ── Handle accessor ──────────────────────────────────────────────────── */
49
50drv_gpio_handle_t *drv_gpio_get_handle(uint8_t dev_id)
51{
52 if (dev_id >= DRV_GPIO_MAX_LINES)
53 return NULL;
54 return &_gpio_handles[dev_id];
55}
