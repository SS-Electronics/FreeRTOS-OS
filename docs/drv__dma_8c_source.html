1/*
2 * drv_dma.c — DMA engine core
3 *
4 * This file is part of FreeRTOS-OS Project.
5 *
6 * Responsibilities
7 * ────────────────
8 * 1. Device registry : dma_register_device()
9 * 2. Channel allocation: dma_request_chan() / dma_release_chan()
10 * 3. Descriptor pool : static per-channel pool of DMA_DESC_POOL_SIZE
11 * 4. Pending queue : dmaengine_submit() / dma_async_issue_pending()
12 * 5. ISR dispatch : dma_complete_callback() / dma_error_callback()
13 *
14 * All HAL-specific work is delegated through dma_chan_hal_ops_t; this file
15 * is vendor-agnostic.

26/* ── Cookie allocator (monotonic, wraps safely) ──────────────────────────── */

28static dma_cookie_t _next_cookie = DMA_MIN_COOKIE;

30static dma_cookie_t _cookie_alloc(void)
31{
32 dma_cookie_t c = _next_cookie++;
33 if (_next_cookie < DMA_MIN_COOKIE)
34 _next_cookie = DMA_MIN_COOKIE;
35 return c;
36}

38/* ── Per-channel descriptor pool ─────────────────────────────────────────── */
39
40static dma_async_tx_descriptor_t *_desc_alloc(dma_chan_t *chan)
41{
42 for (int i = 0; i < DMA_DESC_POOL_SIZE; i++)
43 {
44 if (chan->desc_pool[i].state == DMA_DESC_FREE)
45 {
46 memset(&chan->desc_pool[i], 0, sizeof(dma_async_tx_descriptor_t));
47 chan->desc_pool[i].chan = chan;
48 chan->desc_pool[i].state = DMA_DESC_PREPARED;
49 return &chan->desc_pool[i];
50 }
51 }
52 return NULL;
53}

55static void _desc_free(dma_async_tx_descriptor_t *desc)
56{
57 desc->state = DMA_DESC_FREE;
58}

60/* ── dma_register_device ─────────────────────────────────────────────────── */
61
62int32_t dma_register_device(dma_device_t *dev)
63{
64 if (!dev || !dev->ops || _dev_count >= DMA_MAX_DEVICES)
65 return OS_ERR_OP;
66
67 for (uint8_t i = 0; i < dev->nr_channels && i < DMA_MAX_CHANNELS; i++)
68 {
69 dma_chan_t *ch = &dev->channels[i];
70 ch->device = dev;
71 ch->chan_id = i;
72 ch->state = DMA_CHAN_IDLE;
73 ch->cookie = 0;
74 ch->completed_cookie = 0;
75 ch->active_desc = NULL;
76 INIT_LIST_NODE(&ch->pending);
77
78 for (int j = 0; j < DMA_DESC_POOL_SIZE; j++)
79 ch->desc_pool[j].state = DMA_DESC_FREE;
80
81 if (dev->ops->chan_init)
82 dev->ops->chan_init(ch);
83 }
84
85 list_add_tail(&dev->list, &_dev_list);
86 _dev_count++;
87 return OS_ERR_NONE;
88}

90/* ── dma_request_chan ────────────────────────────────────────────────────── */
91
92dma_chan_t *dma_request_chan(const char *dev_name, uint8_t chan_id)
93{
94 if (!dev_name)
95 return NULL;
96
97 dma_device_t *dev;
98 list_for_each_entry(dev, &_dev_list, list)
99 {
100 if (strcmp(dev->name, dev_name) != 0)
101 continue;
102 if (chan_id >= dev->nr_channels)
103 return NULL;
104
105 dma_chan_t *ch = &dev->channels[chan_id];
106 if (ch->state == DMA_CHAN_IDLE)
107 {
108 ch->state = DMA_CHAN_ALLOCATED;
109 return ch;
110 }
111 }
112 return NULL;
113}

140/* ── Descriptor preparation ──────────────────────────────────────────────── */
141
142dma_async_tx_descriptor_t *dmaengine_prep_slave_single(
143 dma_chan_t *chan, uint32_t buf, uint32_t len,
144 dma_transfer_direction_t dir)
145{
146 if (!chan || chan->state == DMA_CHAN_IDLE || len == 0)
147 return NULL;
148
149 dma_async_tx_descriptor_t *desc = _desc_alloc(chan);
150 if (!desc)
151 return NULL;
152
153 desc->direction = dir;
154 desc->mode = DMA_XFER_NORMAL;
155 desc->len = len;
156
157 if (dir == DMA_MEM_TO_DEV)
158 {
159 desc->src = buf;
160 desc->dst = chan->slave_cfg.dst_addr;
161 }
162 else /* DEV_TO_MEM */
163 {
164 desc->src = chan->slave_cfg.src_addr;
165 desc->dst = buf;
166 }
167 return desc;
168}

201 desc->len = buf_len;

203 desc->src = (dir == DMA_DEV_TO_MEM) ? chan->slave_cfg.src_addr : buf_addr;
204 desc->dst = (dir == DMA_MEM_TO_DEV) ? chan->slave_cfg.dst_addr : buf_addr;
205 return desc;

208/* ── dmaengine_submit ────────────────────────────────────────────────────── */
209
210dma_cookie_t dmaengine_submit(dma_async_tx_descriptor_t *desc)
211{
212 if (!desc || !desc->chan)
213 return (dma_cookie_t)(-OS_ERR_OP);
214
215 desc->cookie = _cookie_alloc();
216 desc->state = DMA_DESC_SUBMITTED;
217 list_add_tail(&desc->node, &desc->chan->pending);
218 desc->chan->cookie = desc->cookie;
219 return desc->cookie;
220}

222/* ── dma_async_issue_pending ─────────────────────────────────────────────── */
223
224void dma_async_issue_pending(dma_chan_t *chan)
225{
226 if (!chan || chan->state == DMA_CHAN_IDLE)
227 return;
228 if (chan->active_desc)
229 return; /* already running — will auto-chain on complete */
230
231 if (list_empty(&chan->pending))
232 return;
233
234 dma_async_tx_descriptor_t *desc = list_first_entry(
235 &chan->pending, dma_async_tx_descriptor_t, node);
236
237 list_delete(&desc->node);
238 desc->state = DMA_DESC_IN_PROGRESS;
239 chan->active_desc = desc;
240 chan->state = DMA_CHAN_BUSY;
241
242 if (chan->device->ops->start)
243 chan->device->ops->start(chan, desc);
244}

246/* ── Control ops ─────────────────────────────────────────────────────────── */
247
248int32_t dmaengine_terminate_all(dma_chan_t *chan)
249{
250 if (!chan)
251 return OS_ERR_OP;
252
253 int32_t ret = OS_ERR_NONE;
254 if (chan->device->ops->terminate_all)
255 ret = chan->device->ops->terminate_all(chan);
256
257 while (!list_empty(&chan->pending))
258 {
259 dma_async_tx_descriptor_t *desc = list_first_entry(
260 &chan->pending, dma_async_tx_descriptor_t, node);
261 list_delete(&desc->node);
262 _desc_free(desc);
263 }
264
265 if (chan->active_desc)
266 {
267 _desc_free(chan->active_desc);
268 chan->active_desc = NULL;
269 }
270
271 if (chan->state == DMA_CHAN_BUSY)
272 chan->state = DMA_CHAN_ALLOCATED;
273
274 return ret;
275}

298/* ── ISR dispatch — called by HAL backend from interrupt context ─────────── */
299
300void dma_complete_callback(dma_chan_t *chan)
301{
302 if (!chan)
303 return;
304
305 dma_async_tx_descriptor_t *desc = chan->active_desc;
306 if (!desc)
307 return;
308
309 chan->completed_cookie = desc->cookie;
310
311 if (desc->callback)
312 desc->callback(desc->callback_param);
313
314 /* Cyclic: keep descriptor active; HAL re-arms automatically */
315 if (desc->mode == DMA_XFER_CYCLIC)
316 {
317 desc->state = DMA_DESC_IN_PROGRESS;
318 return;
319 }
320
321 desc->state = DMA_DESC_COMPLETE;
322 _desc_free(desc);
323 chan->active_desc = NULL;
324 chan->state = DMA_CHAN_ALLOCATED;
325
326 /* Auto-start next queued descriptor */
327 dma_async_issue_pending(chan);
328}

330void dma_error_callback(dma_chan_t *chan)
331{
332 if (!chan)
333 return;
334
335 dma_async_tx_descriptor_t *desc = chan->active_desc;
336 if (!desc)
337 return;
338
339 if (desc->error_callback)
340 desc->error_callback(desc->callback_param);
341
342 desc->state = DMA_DESC_ERROR;
343 _desc_free(desc);
344 chan->active_desc = NULL;
345 chan->state = DMA_CHAN_ALLOCATED;
346}
