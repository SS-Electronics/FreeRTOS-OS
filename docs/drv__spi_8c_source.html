23int32_t drv_spi_register(uint8_t dev_id,
24 const drv_spi_hal_ops_t *ops,
25 uint32_t clock_hz,
26 uint32_t timeout_ms)
27{
28 if (dev_id >= BOARD_SPI_COUNT || ops == NULL)
29 return OS_ERR_OP;
30
31 drv_spi_handle_t *h = &_spi_handles[dev_id];
32
33 h->dev_id = dev_id;
34 h->ops = ops;
35 h->clock_hz = clock_hz;
36 h->timeout_ms = timeout_ms;
37 h->initialized = 0;
38 h->last_err = OS_ERR_NONE;
39
40 return h->ops->hw_init(h);
41}

54int32_t drv_spi_transmit(uint8_t dev_id, const uint8_t *data, uint16_t len)
55{
56 if (dev_id >= BOARD_SPI_COUNT || data == NULL)
57 return OS_ERR_OP;
58
59 drv_spi_handle_t *h = &_spi_handles[dev_id];
60
61 if (!h->initialized || h->ops == NULL)
62 return OS_ERR_OP;
63
64 return h->ops->transmit(h, data, len, h->timeout_ms);
65}

80int32_t drv_spi_transfer(uint8_t dev_id,
