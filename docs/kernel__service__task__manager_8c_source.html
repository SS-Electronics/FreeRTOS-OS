1/*
2 * kernel_service_task_manager.c — Background task health monitor
3 *
4 * This file is part of FreeRTOS-OS Project.
5 *
6 * Responsibilities
7 * ────────────────
8 * 1. Periodic health snapshot: walks thread_list and collects per-task
9 * state + stack high-watermark via FreeRTOS trace APIs.
10 * 2. Timer-daemon monitoring: queries the FreeRTOS timer task handle.
11 * 3. Heap health: free / minimum-ever heap tracked each scan cycle.
12 * 4. FreeRTOS application hooks:
13 * vApplicationStackOverflowHook — configCHECK_FOR_STACK_OVERFLOW 2
14 * vApplicationIdleHook — configUSE_IDLE_HOOK 1
15 * vApplicationMallocFailedHook — configUSE_MALLOC_FAILED_HOOK 1
16 *
17 * Configuration requirements in FreeRTOSConfig.h
18 * ───────────────────────────────────────────────
19 * configUSE_IDLE_HOOK 1
20 * configCHECK_FOR_STACK_OVERFLOW 2
21 * configUSE_MALLOC_FAILED_HOOK 1
22 * configUSE_TRACE_FACILITY 1 (already set)
23 * INCLUDE_uxTaskGetStackHighWaterMark 1 (already set)
24 * INCLUDE_eTaskGetState 1 (already set)
25 */

46/* ── Internal: collect one task's metrics ─────────────────────────────────── */
47
48static void _collect_task(task_health_t *out, TaskHandle_t h, uint32_t tid)
49{
50 out->handle = h;
51 out->name = pcTaskGetName(h);
52 out->state = eTaskGetState(h);
53 out->stack_hwm = uxTaskGetStackHighWaterMark(h);
54 out->thread_id = tid;
55}

57/* ── Internal: scan the entire thread_list ────────────────────────────────── */
58
59static void _scan_tasks(void)
60{
61 _health.task_count = 0;
62
63 thread_handle_t *pos;
64 list_for_each_entry(pos, os_thread_list_get(), list)
65 {
66 if (_health.task_count >= TASK_MGR_MAX_TASKS)
67 break;
68
69 _collect_task(&_health.tasks[_health.task_count],
70 pos->thread_handle,
71 pos->thread_id);
72 _health.task_count++;
73 }
74}

76/* ── Internal: collect timer-daemon metrics ───────────────────────────────── */
77
78static void _scan_timer_task(void)
79{
80#if (configUSE_TIMERS == 1)

82 if (h != NULL)
83 {
84 _collect_task(&_health.timer_task, h, 0);

87 else
88#endif

90 _health.timer_task_valid = 0;

109/* ── FreeRTOS application hooks ───────────────────────────────────────────── */

111/*
112 * Called by the idle task each time it runs (configUSE_IDLE_HOOK 1).
113 * Must never attempt to block or call any API that might block.
114 */
115void vApplicationIdleHook(void)
116{
117 _idle_tick_count++;
118}
119
120/*
121 * Called when a task overflows its stack (configCHECK_FOR_STACK_OVERFLOW 2).
122 * Spin forever — the application is in an unrecoverable state.
123 * pxCurrentTCB / pcTaskName identify the offending task for a debugger.
124 */
125void vApplicationStackOverflowHook(TaskHandle_t xTask, char *pcTaskName)
126{
127 (void)xTask;
128 (void)pcTaskName;
129 _stack_overflow_count++;
130 taskDISABLE_INTERRUPTS();
131 for (;;) {}
132}
133
134/*
135 * Called when pvPortMalloc() fails (configUSE_MALLOC_FAILED_HOOK 1).
136 * Increment counter so the health task can report it; do not block.
137 */
138void vApplicationMallocFailedHook(void)
139{
140 _malloc_fail_count++;
141}

143/* ── Background monitor task ──────────────────────────────────────────────── */
144
145void thread_task_mgmt(void *arg)
146{
147 (void)arg;
148
149 /* One-shot startup delay so all service threads have registered first */
150 vTaskDelay(pdMS_TO_TICKS(TASK_MGR_SCAN_PERIOD_MS));
151
152 while (1)
153 {
154 _scan_tasks();
155 _scan_timer_task();
156 _scan_heap();
157
158 _health.stack_overflow_count = _stack_overflow_count;
159 _health.malloc_fail_count = _malloc_fail_count;
160
161 vTaskDelay(pdMS_TO_TICKS(TASK_MGR_SCAN_PERIOD_MS));
162 }
163}
