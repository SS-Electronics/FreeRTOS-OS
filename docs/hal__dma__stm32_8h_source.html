1/*
2 * hal_dma_stm32.h — STM32F4 DMA HAL backend for the DMA engine
3 *
4 * This file is part of FreeRTOS-OS Project.
5 *
6 * Exports two pre-built dma_device_t instances (one per controller):
7 *
8 * hal_dma1_device — DMA1, streams 0–7
9 * hal_dma2_device — DMA2, streams 0–7
10 *
11 * Call hal_dma_stm32_init() once at boot (after clocks are on) before
12 * registering them with the DMA engine. hal_dma_stm32_init() internally
13 * calls dma_register_device() for both controllers.
14 *
15 * STM32F411 stream → peripheral request mapping (DMA channel mux):
16 * DMA1 Stream 5, Channel 4 → USART2_RX
17 * DMA1 Stream 6, Channel 4 → USART2_TX
18 * DMA1 Stream 0, Channel 1 → I2C1_RX
19 * DMA1 Stream 6, Channel 1 → I2C1_TX
20 * DMA2 Stream 3, Channel 3 → SPI1_RX
21 * DMA2 Stream 3, Channel 3 → SPI1_TX
22 */
