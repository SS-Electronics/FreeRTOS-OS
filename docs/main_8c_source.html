89/* ── OS entry point ───────────────────────────────────────────────────────── */
90
91__SECTION_BOOT __USED void main(void)
92{
93 /* 1. HAL init and system clock. All peripheral clocks are derived from
94 * SYSCLK so this must complete before any peripheral is touched. */
95 HAL_Init();
96 drv_rcc_clock_init();
97
98 /* 1a. Enable all RCC clocks (sys bus, peripherals, GPIO ports) so every
99 * HAL_xxx_MspInit() can configure GPIO and NVIC without enabling clocks
100 * itself. Must run before irq_hw_init_all() and before any peripheral
101 * management thread calls HAL_xxx_Init(). */
102 board_clk_enable();
103
104 /* 2. Configure NVIC group bits required by FreeRTOS before any
105 * interrupt-driven peripheral is initialised. */
106 drv_cpu_interrupt_prio_set();
107
108 /* 2a. Initialise irq_desc table and bind irq_chip to all hardware-backed
109 * software IRQ IDs. Must run before any request_irq() call.
110 * TIM1 (HAL tick) is already firing at this point; hal_timebase_stm32_irq_handler
111 * uses an early-boot fallback until handle_irq is populated here. */
112 irq_hw_init_all();
113
114 /* 3. IPC ring-buffer queues — UART mgmt reads ring-buffer handles after
115 * its startup delay, so these must be allocated first. */
116 ipc_queues_init();
117
118 /* 4. Register all OS service tasks in one call.
119 * See services/kernel_service_core.c for the full registration table. */
120 os_kernel_thread_register();
121
122 /* 5. Bind printk() to the TX ring-buffer of COMM_PRINTK_HW_ID.
123 * Must be called after ipc_queues_init() registers the queue. */
124 printk_init();
125
126 /* 6. Application layer. app_main() creates tasks, optionally registers
127 * additional IRQ subscribers via irq_register(), then returns.
128 * Tasks are driven by the scheduler; app_main must not block. */
129 app_main();
130
131 /* 7. Hand over to FreeRTOS. Interrupts are enabled for the first time
132 * inside vPortSVCHandler when the first task context loads. */
133 vTaskStartScheduler();
134
135 while (1) {}
136}
