16int32_t uart_sync_transmit(uint8_t dev_id, const uint8_t *data,
17 uint16_t len, uint32_t timeout_ms)
18{
19 QueueHandle_t q = uart_mgmt_get_queue();
20 if (q == NULL || data == NULL || len == 0)
21 return OS_ERR_OP;
22
23 int32_t result = OS_ERR_OP;
24
25 uart_mgmt_msg_t msg = {
26 .cmd = UART_MGMT_CMD_TRANSMIT,
27 .dev_id = dev_id,
28 .data = data,
29 .len = len,
30 .result_notify = xTaskGetCurrentTaskHandle(),
31 .result_code = &result,
32 };
33
34 if (xQueueSend(q, &msg, pdMS_TO_TICKS(timeout_ms)) != pdTRUE)
35 return OS_ERR_OP;
36
37 ulTaskNotifyTake(pdTRUE, pdMS_TO_TICKS(timeout_ms));
38 return result;
39}

41int32_t uart_sync_receive(uint8_t dev_id, uint8_t *buf,
42 uint16_t len, uint32_t timeout_ms)
43{
44 if (buf == NULL || len == 0)
45 return OS_ERR_OP;
46
47 uint32_t elapsed = 0;
48 uint16_t received = 0;
49
50 while (received < len)
51 {
52 if (uart_mgmt_read_byte(dev_id, &buf[received]) == OS_ERR_NONE)
53 {
54 received++;
55 }
56 else
57 {
58 if (elapsed >= timeout_ms)
59 return OS_ERR_OP;
60 os_thread_delay(1);
61 elapsed++;
62 }
63 }
64
65 return OS_ERR_NONE;
66}
