1/*
2 * kernel_service_core.c — Central OS service task registration
3 *
4 * This file is part of FreeRTOS-OS Project.
5 *
6 * All FreeRTOS service threads are registered here via a single call to
7 * os_kernel_thread_register() from main(). Each service's _start() function
8 * is responsible for its own internal queue creation and task spawn — this
9 * file is only the dispatch table.
10 *
11 * Registration order
12 * ──────────────────
13 * 1. task_mgr — health monitor (always enabled)
14 * 2. gpio_mgmt — GPIO command queue (always enabled)
15 * 3. uart_mgmt — UART RX/TX daemon (CONFIG_INC_SERVICE_UART_MGMT)
16 * 4. iic_mgmt — I2C transfer daemon (INC_SERVICE_IIC_MGMT)
17 * 5. spi_mgmt — SPI transfer daemon (BOARD_SPI_COUNT > 0)
18 */

37status_t os_kernel_thread_register(void)
38{
39 status_t status = OS_ERR_NONE;
40
41 /* Task health monitor */
42 if (os_thread_create(thread_task_mgmt, "task_mgr",
43 PROC_SERVICE_TASK_MGMT_STACK_SIZE,
44 PROC_SERVICE_TASK_MGMT_PRIORITY, NULL) < 0)
45 status |= OS_ERR_OP;
46
47 /* GPIO management */
48 if (gpio_mgmt_start() != OS_ERR_NONE)
49 status |= OS_ERR_OP;
50
51#if (INC_SERVICE_UART_MGMT == 1)
52 if (uart_mgmt_start() != OS_ERR_NONE)
53 status |= OS_ERR_OP;
54#endif
55
56#if (INC_SERVICE_IIC_MGMT == 1)
57 if (iic_mgmt_start() != OS_ERR_NONE)
58 status |= OS_ERR_OP;
59#endif
60
61#if (BOARD_SPI_COUNT > 0)
62 if (spi_mgmt_start() != OS_ERR_NONE)
63 status |= OS_ERR_OP;
64#endif
65
66 /* Start interactive shell on UART_APP (USART2, PA2/PA3, /dev/ttyUSB0).
67 * The shell task delays TIME_OFFSET_OS_SHELL_MGMT ms internally so it
68 * waits for uart_mgmt to complete UART initialisation first. */
69 os_shell_mgmt_start();
70
71 return status;
72}
