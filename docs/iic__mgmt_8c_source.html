1/*
2 * iic_mgmt.c — I2C management service thread
3 *
4 * This file is part of FreeRTOS-OS Project.
5 *
6 * The management thread serialises I2C operations onto each bus. All
7 * transfers use the interrupt-driven HAL ops (transmit_it / receive_it) so
8 * the thread releases the CPU while the hardware works. Completion is
9 * signalled from the ISR callback via the generic IRQ notification framework:
10 *
11 * mgmt thread ISR (HAL_I2C_*CpltCallback)
12 * ────────── ────────────────────────────
13 * h->notify_task = currentTask()
14 * ops->transmit_it(...) ──────────► irq_notify_from_isr(IRQ_ID_I2C_TX_DONE)
15 * ulTaskNotifyTake(...) ◄────────── vTaskNotifyGiveFromISR(h->notify_task)
16 * result = h->last_err
17 */
18

58static void iic_mgmt_thread(void *arg)
59{
60 (void)arg;
61
62 os_thread_delay(TIME_OFFSET_IIC_MANAGEMENT);
63
64 /* Register all I2C buses described in the board configuration */
65 {
66 const drv_iic_hal_ops_t *ops =
67#if (CONFIG_DEVICE_VARIANT == MCU_VAR_STM)
68 hal_iic_stm32_get_ops();
69#elif (CONFIG_DEVICE_VARIANT == MCU_VAR_INFINEON)
70 hal_iic_infineon_get_ops();
71#else
72 NULL;
73#endif
74 if (ops != NULL)
75 {
76 const board_config_t *bc = board_get_config();
77 for (uint8_t i = 0; i < bc->iic_count; i++)
78 {
79 const board_iic_desc_t *d = &bc->iic_table[i];
80 drv_iic_handle_t *h = drv_iic_get_handle(d->dev_id);
81
82#if (CONFIG_DEVICE_VARIANT == MCU_VAR_STM)
83 hal_iic_stm32_set_config(h, d->instance,
84 d->addr_mode, d->dual_addr);
85#endif
86 int32_t err = drv_iic_register(d->dev_id, ops,
87 d->clock_hz, IIC_ACK_TIMEOUT_MS);
88 (void)err;
89
90 /* Subscribe for IT completion and error notifications */
91 irq_register(IRQ_ID_I2C_TX_DONE(d->dev_id), _iic_done_cb, h);
92 irq_register(IRQ_ID_I2C_RX_DONE(d->dev_id), _iic_done_cb, h);
93 irq_register(IRQ_ID_I2C_ERROR(d->dev_id), _iic_err_cb, h);
94 }
95 }
96 }
97
98 iic_mgmt_msg_t msg;
99
100 for (;;)
101 {
102 if (xQueueReceive(_mgmt_queue, &msg, portMAX_DELAY) != pdTRUE)
103 continue;
104
105 drv_iic_handle_t *h = drv_iic_get_handle(msg.bus_id);
106 if (h == NULL || h->ops == NULL)
107 goto notify;
108
109 int32_t result = OS_ERR_OP;
110
111 switch (msg.cmd)
112 {
113 case IIC_MGMT_CMD_TRANSMIT:
114 if (h->ops->transmit_it != NULL)
115 {
116 h->notify_task = xTaskGetCurrentTaskHandle();
117 result = h->ops->transmit_it(h, msg.dev_addr, msg.reg_addr,
118 msg.use_reg, msg.data, msg.len);
119 if (result == OS_ERR_NONE)
120 ulTaskNotifyTake(pdTRUE, pdMS_TO_TICKS(h->timeout_ms));
121 result = h->last_err;
122 h->notify_task = NULL;
123 }
124 else
125 {
126 result = h->ops->transmit(h, msg.dev_addr, msg.reg_addr,
127 msg.use_reg, msg.data, msg.len,
128 h->timeout_ms);
129 }
130 break;
131
132 case IIC_MGMT_CMD_RECEIVE:
133 if (h->ops->receive_it != NULL)
134 {
135 h->notify_task = xTaskGetCurrentTaskHandle();
136 result = h->ops->receive_it(h, msg.dev_addr, msg.reg_addr,
137 msg.use_reg, msg.data, msg.len);
138 if (result == OS_ERR_NONE)
139 ulTaskNotifyTake(pdTRUE, pdMS_TO_TICKS(h->timeout_ms));
140 result = h->last_err;
141 h->notify_task = NULL;
142 }
143 else
144 {
145 result = h->ops->receive(h, msg.dev_addr, msg.reg_addr,
146 msg.use_reg, msg.data, msg.len,
147 h->timeout_ms);
148 }
149 break;
150
151 case IIC_MGMT_CMD_PROBE:
152 /* HAL has no IT variant for IsDeviceReady — use blocking */
153 result = h->ops->is_device_ready(h, msg.dev_addr, h->timeout_ms);
154 break;
155
156 case IIC_MGMT_CMD_REINIT:
157 h->ops->hw_deinit(h);
158 result = h->ops->hw_init(h);
159 break;
160
161 default:
162 break;
163 }
164
165 if (msg.result_code != NULL)
166 *msg.result_code = result;
167
168notify:
169 if (msg.result_notify != NULL)
170 xTaskNotifyGive(msg.result_notify);
171 }
172}

174int32_t iic_mgmt_start(void)
175{
176 _mgmt_queue = xQueueCreate(IIC_MGMT_QUEUE_DEPTH, sizeof(iic_mgmt_msg_t));
177 if (_mgmt_queue == NULL)
178 return OS_ERR_MEM_OF;
179
180 int32_t tid = os_thread_create(iic_mgmt_thread,
181 "iic_mgmt",
182 PROC_SERVICE_IIC_MGMT_STACK_SIZE,
183 PROC_SERVICE_IIC_MGMT_PRIORITY,
184 NULL);
185 return (tid >= 0) ? OS_ERR_NONE : OS_ERR_OP;
186}

193int32_t iic_mgmt_async_transmit(uint8_t bus_id, uint16_t dev_addr,
194 uint8_t reg_addr, uint8_t use_reg,
195 const uint8_t *data, uint16_t len)
196{
197 if (_mgmt_queue == NULL || data == NULL)
198 return OS_ERR_OP;
199
200 iic_mgmt_msg_t msg = {
201 .cmd = IIC_MGMT_CMD_TRANSMIT,
202 .bus_id = bus_id,
203 .dev_addr = dev_addr,
204 .reg_addr = reg_addr,
205 .use_reg = use_reg,
206 .data = (uint8_t *)data,
207 .len = len,
208 .result_notify = NULL,
209 .result_code = NULL,
210 };
211
212 return (xQueueSend(_mgmt_queue, &msg, 0) == pdTRUE) ? OS_ERR_NONE : OS_ERR_OP;
213}
