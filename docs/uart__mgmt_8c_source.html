1/*
2 * uart_mgmt.c — UART management service thread
3 *
4 * This file is part of FreeRTOS-OS Project.
5 *
6 * IRQ flow (chip → app)
7 * ─────────────────────
8 *
9 * [startup_stm32f411vetx.c] vector table — weak alias USART1_IRQHandler
10 * │ overridden by strong symbol in…
11 * ▼
12 * [irq_periph_dispatch_generated.c] USART1_IRQHandler() { hal_uart_stm32_irq_handler(USART1); }
13 * │
14 * ▼
15 * [hal_uart_stm32.c] HAL_UART_IRQHandler() → HAL_UART_RxCpltCallback()
16 * │ HAL_UART_TxCpltCallback()
17 * │ HAL_UART_ErrorCallback()
18 * │
19 * ▼ irq_notify_from_isr(IRQ_ID_UART_RX(n), &byte, pxHPT)
20 * [irq/irq.c] generic pub/sub dispatch
21 * │
22 * ├──► _uart_rx_cb (this file, ISR context)
23 * │ → ringbuffer_putchar(rb, byte)
24 * │
25 * └──► app-registered callbacks (e.g. echo_task wakeup)
26 * → vTaskNotifyGiveFromISR(app_task, pxHPT)
27 *
28 * Thread lifecycle
29 * ────────────────
30 * 1. uart_mgmt_start() creates the FreeRTOS task and management queue.
31 * 2. After TIME_OFFSET_SERIAL_MANAGEMENT ms the task registers every
32 * enabled UART with the generic driver layer and subscribes to the
33 * generic IRQ notification framework for RX, TX-done, and error events.
34 * 3. The task then loops on the management queue, processing transmit /
35 * reinit / deinit commands.
36 */

63/*
64 * _uart_rx_cb — byte received on a UART channel.
65 * arg = struct ringbuffer * for the channel's RX queue.
66 * data = uint8_t * to the received byte (lives in the HAL staging array).

69 void *data,

97/* ── Management thread ────────────────────────────────────────────────────── */
98
99static void uart_mgmt_thread(void *arg)
100{
101 (void)arg;
102
103 os_thread_delay(TIME_OFFSET_SERIAL_MANAGEMENT);
104
105 /* ── Register and initialise all enabled UART devices ── */
106 {
107 const drv_uart_hal_ops_t *ops =
108#if (CONFIG_DEVICE_VARIANT == MCU_VAR_STM)
109 hal_uart_stm32_get_ops();
110#elif (CONFIG_DEVICE_VARIANT == MCU_VAR_INFINEON)
111 hal_uart_infineon_get_ops();
112#else
113 NULL;
114#endif
115 if (ops != NULL)
116 {
117 const board_config_t *bc = board_get_config();
118
119 for (uint8_t i = 0; i < bc->uart_count; i++)
120 {
121 const board_uart_desc_t *d = &bc->uart_table[i];
122 drv_uart_handle_t *h = drv_uart_get_handle(d->dev_id);
123
124#if (CONFIG_DEVICE_VARIANT == MCU_VAR_STM)
125 hal_uart_stm32_set_config(h, d->instance,
126 d->word_len, d->stop_bits,
127 d->parity, d->mode);
128#endif
129 int32_t err = drv_uart_register(d->dev_id, ops,
130 d->baudrate, 10);
131 (void)err;
132
133 /*
134 * Subscribe to the generic IRQ notification framework.
135 *
136 * _uart_rx_cb feeds every received byte straight into the
137 * channel's RX ring buffer, which callers drain via
138 * uart_mgmt_read_byte(). The arg pointer is retrieved here
139 * after ipc_queues_init() has already allocated the buffer.
140 */
141 struct ringbuffer *rb = (struct ringbuffer *)
142 ipc_mqueue_get_handle(
143 global_uart_rx_mqueue_list[d->dev_id]);
144
145 irq_register(IRQ_ID_UART_RX(d->dev_id), _uart_rx_cb, rb);
146 irq_register(IRQ_ID_UART_ERROR(d->dev_id), _uart_err_cb, h);
147 }
148 }
149 }
150
151 /* ── Main event loop ── */
152 uart_mgmt_msg_t msg;
153
154 for (;;)
155 {
156 if (xQueueReceive(_mgmt_queue, &msg, portMAX_DELAY) != pdTRUE)
157 continue;
158
159 drv_uart_handle_t *h = drv_uart_get_handle(msg.dev_id);
160 if (h == NULL)
161 continue;
162
163 int32_t result = OS_ERR_OP;
164
165 switch (msg.cmd)
166 {
167 case UART_MGMT_CMD_TRANSMIT:
168 if (msg.data != NULL && msg.len > 0)
169 {
170 result = h->ops->transmit(h, msg.data, msg.len, h->timeout_ms);
171 if (result != OS_ERR_NONE)
172 {
173 h->ops->hw_deinit(h);
174 h->ops->hw_init(h);
175 }
176 }
177 break;
178
179 case UART_MGMT_CMD_REINIT:
180 h->ops->hw_deinit(h);
181 result = h->ops->hw_init(h);
182 break;
183
184 case UART_MGMT_CMD_DEINIT:
185 h->ops->hw_deinit(h);
186 result = OS_ERR_NONE;
187 break;
188
189 default:
190 break;
191 }
192
193 if (msg.result_code != NULL)
194 *msg.result_code = result;
195 if (msg.result_notify != NULL)
196 xTaskNotifyGive(msg.result_notify);
197 }
198}

202int32_t uart_mgmt_start(void)
203{
204 _mgmt_queue = xQueueCreate(UART_MGMT_QUEUE_DEPTH, sizeof(uart_mgmt_msg_t));
205 if (_mgmt_queue == NULL)
206 return OS_ERR_MEM_OF;
207
208 int32_t tid = os_thread_create(uart_mgmt_thread,
209 "uart_mgmt",
210 PROC_SERVICE_SERIAL_MGMT_STACK_SIZE,
211 PROC_SERVICE_SERIAL_MGMT_PRIORITY,
212 NULL);
213 return (tid >= 0) ? OS_ERR_NONE : OS_ERR_OP;
214}

221int32_t uart_mgmt_async_transmit(uint8_t dev_id, const uint8_t *data, uint16_t len)
222{
223 if (_mgmt_queue == NULL || data == NULL || len == 0)
224 return OS_ERR_OP;
225
226 uart_mgmt_msg_t msg = {
227 .cmd = UART_MGMT_CMD_TRANSMIT,
228 .dev_id = dev_id,
229 .data = data,
230 .len = len,
231 .result_notify = NULL,
232 .result_code = NULL,
233 };
234
235 BaseType_t ok = xQueueSend(_mgmt_queue, &msg, 0);
236 return (ok == pdTRUE) ? OS_ERR_NONE : OS_ERR_OP;
237}

239int32_t uart_mgmt_read_byte(uint8_t dev_id, uint8_t *byte)
240{
241 if (dev_id >= NO_OF_UART || byte == NULL)
242 return OS_ERR_OP;
243
244 struct ringbuffer *rb =
245 (struct ringbuffer *)ipc_mqueue_get_handle(
246 global_uart_rx_mqueue_list[dev_id]);
247
248 if (rb == NULL)
249 return OS_ERR_OP;
250
251 return (ringbuffer_getchar(rb, byte) > 0U) ? OS_ERR_NONE : OS_ERR_OP;
252}
