1/*
2 * os_shell_management.c — FreeRTOS+CLI interactive shell service
3 *
4 * This file is part of FreeRTOS-OS Project.
5 *
6 * Shell I/O wiring
7 * ────────────────
8 * RX: UART_SHELL (UART_APP, USART2) RX ring buffer — filled by the RXNE ISR
9 * via drv_uart_rx_isr_dispatch() → _uart_rx_cb (uart_mgmt.c).
10 * The shell task polls the ring buffer with a short yield when empty.
11 *
12 * TX: Written directly to the UART_SHELL TX ring buffer via
13 * ringbuffer_put(), then drv_uart_tx_kick() enables TXEIE so the
14 * existing TXE ISR drains the buffer byte by byte.
15 *
16 * Line editing
17 * ────────────
18 * • Printable characters are echoed and buffered.
19 * • BS (0x08) and DEL (0x7F) erase the last character.
20 * • CR / LF submits the line to FreeRTOS_CLIProcessCommand().
21 * • Lines longer than SHELL_LINE_BUF_LEN-1 bytes are silently truncated.
22 *
23 * FreeRTOS+CLI output loop
24 * ────────────────────────
25 * FreeRTOS_CLIProcessCommand() is called repeatedly until it returns
26 * pdFALSE (no more output pending) so multi-page command output works.
27 * Each chunk is written to the TX ring buffer before the next call.
28 */

90/* ── TX helper — writes directly to the shell UART TX ring buffer ──────────
91 *
92 * Bypasses the uart_mgmt queue so the interrupt-driven TX path (TXEIE ISR +
93 * ring buffer, set up by uart_mgmt at startup) handles transmission.
94 * This is the same mechanism printk() uses.
95 */
96static void _shell_write(const char *str, uint16_t len)
97{
98 if (str == NULL || len == 0)
99 return;
100
101 struct ringbuffer *rb = (struct ringbuffer *)
102 ipc_mqueue_get_handle(global_uart_tx_mqueue_list[UART_SHELL_HW_ID]);
103
104 if (rb == NULL)
105 return;
106
107 for (uint16_t i = 0; i < len; i++)
108 ringbuffer_putchar(rb, (uint8_t)str[i]);
109
110 drv_uart_tx_kick(UART_SHELL_HW_ID);
111}

154/* ── Line processor ───────────────────────────────────────────────────────── */
155
156static void _shell_process_line(void)
157{
158 BaseType_t more;
159
160 if (_line_pos == 0)
161 {
162 _shell_write(SHELL_PROMPT, (uint16_t)(sizeof(SHELL_PROMPT) - 1));
163 return;
164 }
165
166 _shell_write("\r\n", 2);
167
168 do {
169 memset(_out_buf, 0, sizeof(_out_buf));
170 more = FreeRTOS_CLIProcessCommand(_line_buf, _out_buf, sizeof(_out_buf));
171 uint16_t olen = (uint16_t)strlen(_out_buf);
172 if (olen > 0)
173 _shell_write(_out_buf, olen);
174 } while (more == pdTRUE);
175
176 _shell_write(SHELL_PROMPT, (uint16_t)(sizeof(SHELL_PROMPT) - 1));
177
178 memset(_line_buf, 0, sizeof(_line_buf));
179 _line_pos = 0;
180}

182/* ── Shell task ───────────────────────────────────────────────────────────── */
183
184static void _os_shell_task(void *arg)
185{
186 (void)arg;
187
188 /* Wait for UART_SHELL to be registered by uart_mgmt */
189 os_thread_delay(TIME_OFFSET_OS_SHELL_MGMT);
190
191 /* Register built-in commands */
192 FreeRTOS_CLIRegisterCommand(&_cmd_help);
193 FreeRTOS_CLIRegisterCommand(&_cmd_version);
194 FreeRTOS_CLIRegisterCommand(&_cmd_uptime);
195 FreeRTOS_CLIRegisterCommand(&_cmd_reboot);
196
197 /* Register shell command modules */
198 shell_task_mgmt_register_cmds();
199
200 /* Print welcome banner */
201 _print_banner();
202 _shell_write(SHELL_PROMPT, (uint16_t)(sizeof(SHELL_PROMPT) - 1));
203
204 memset(_line_buf, 0, sizeof(_line_buf));
205 _line_pos = 0;
206
207 for (;;)
208 {
209 uint8_t byte;
210
211 /* Poll the UART_SHELL RX ring buffer; yield when empty */
212 struct ringbuffer *rx_rb = (struct ringbuffer *)
213 ipc_mqueue_get_handle(global_uart_rx_mqueue_list[UART_SHELL_HW_ID]);
214
215 if (rx_rb == NULL || ringbuffer_getchar(rx_rb, &byte) == 0)
216 {
217 vTaskDelay(pdMS_TO_TICKS(5));
218 continue;
219 }
220
221 switch (byte)
222 {
223 case '\r':
224 case '\n':
225 _line_buf[_line_pos] = '\0';
226 _shell_process_line();
227 break;
228
229 case '\b': /* BS */
230 case 0x7FU: /* DEL */
231 if (_line_pos > 0)
232 {
233 _line_pos--;
234 _line_buf[_line_pos] = '\0';
235 _shell_write("\b \b", 3);
236 }
237 break;
238
239 case 0x1BU: /* ESC — disable printk debug output */
240 printk_disable();
241 _shell_write("\r\n[debug off]\r\n", 15);
242 _shell_write(SHELL_PROMPT, (uint16_t)(sizeof(SHELL_PROMPT) - 1));
243 break;
244
245 case 0x03U: /* Ctrl-C — discard current line */
246 memset(_line_buf, 0, sizeof(_line_buf));
247 _line_pos = 0;
248 _shell_write("^C", 2);
249 _shell_write(SHELL_PROMPT, (uint16_t)(sizeof(SHELL_PROMPT) - 1));
250 break;
251
252 default:
253 if (byte >= 0x20U && _line_pos < (SHELL_LINE_BUF_LEN - 1U))
254 {
255 _line_buf[_line_pos++] = (char)byte;
256 _shell_write((const char *)&byte, 1); /* local echo */
257 }
258 break;
259 }
260 }
261}

297{
298 (void)in;

300 /* Flush output before reset */

302 vTaskDelay(pdMS_TO_TICKS(50));
303 NVIC_SystemReset();
304 return pdFALSE; /* unreachable */
