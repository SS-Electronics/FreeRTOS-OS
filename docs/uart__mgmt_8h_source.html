1/*
2 * uart_mgmt.h — UART management service thread
3 *
4 * This file is part of FreeRTOS-OS Project.
5 *
6 * The UART management thread owns all drv_uart_handle_t instances for the
7 * system. It:
8 * • Registers and initialises each enabled UART peripheral at startup.
9 * • Accepts asynchronous transmit requests via a FreeRTOS queue.
10 * • Monitors the last_err field and attempts peripheral recovery on error.
11 *
12 * Usage
13 * ─────
14 * Include this header and call uart_mgmt_start() from the OS init sequence.
15 * Upper layers post uart_mgmt_msg_t messages to the management queue
16 * (uart_mgmt_get_queue()) to request transmissions without blocking the
17 * calling thread on the UART bus.
18 */
19

36typedef enum {
37 UART_MGMT_CMD_TRANSMIT = 0, /**< Async transmit request */
38 UART_MGMT_CMD_REINIT, /**< Re-initialise peripheral after error */
39 UART_MGMT_CMD_DEINIT, /**< Graceful shutdown of a UART device */
40} uart_mgmt_cmd_t;

42/**
43 * @struct uart_mgmt_msg_t
44 * @brief Message posted to the management thread queue.
45 *
46 * For UART_MGMT_CMD_TRANSMIT: fill dev_id, data pointer, and len.
47 * The data buffer must remain valid until the management thread drains the
48 * message (use a static buffer or a heap allocation the caller frees later).
49 */
50typedef struct {
51 uart_mgmt_cmd_t cmd;
52 uint8_t dev_id;
53 const uint8_t *data;
54 uint16_t len;
55 TaskHandle_t result_notify; /**< Task to notify on completion (NULL = async) */
56 int32_t *result_code; /**< Where to store the return code (NULL = ignore) */
57} uart_mgmt_msg_t;

71/**
72 * @brief Create and start the UART management thread.
73 *
74 * Call this once from the OS init sequence, before the FreeRTOS scheduler
75 * starts. The thread will perform hardware initialisation after the
76 * scheduler starts (after the startup delay defined in os_config.h).
77 *
78 * @return OS_ERR_NONE on success, negative OS_ERR_* on failure.
