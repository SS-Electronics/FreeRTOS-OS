29static int32_t stm32_iic_hw_init(drv_iic_handle_t *h)
30{
31 if (h == NULL)
32 return OS_ERR_OP;
33
34 I2C_HandleTypeDef *hi2c = &h->hw.hi2c;
35 const board_iic_desc_t *d = board_find_iic(hi2c->Instance);
36 if (d == NULL)
37 return OS_ERR_OP;
38
39 /* GPIO and NVIC setup — owned here so HAL_I2C_MspInit (called internally
40 * by HAL_I2C_Init below) is a no-op and init ownership stays in this fn. */
41 GPIO_InitTypeDef gpio = {
42 .Mode = d->scl_pin.mode,
43 .Pull = d->scl_pin.pull,
44 .Speed = d->scl_pin.speed,
45 .Alternate = d->scl_pin.alternate,
46 };
47
48 gpio.Pin = d->scl_pin.pin;
49 HAL_GPIO_Init(d->scl_pin.port, &gpio);
50
51 gpio.Pin = d->sda_pin.pin;
52 gpio.Alternate = d->sda_pin.alternate;
53 HAL_GPIO_Init(d->sda_pin.port, &gpio);
54
55 drv_irq_enable((int32_t)d->ev_irqn, d->irq_priority);
56 drv_irq_enable((int32_t)d->er_irqn, d->irq_priority);
57
58 hi2c->Init.ClockSpeed = h->clock_hz;
59
60 HAL_StatusTypeDef ret = HAL_I2C_Init(hi2c);
61 if (ret == HAL_OK)
62 h->initialized = 1;
63
64 h->last_err = _hal_to_os_err(ret);
65 return h->last_err;
66}

68static int32_t stm32_iic_hw_deinit(drv_iic_handle_t *h)
69{
70 if (h == NULL)
71 return OS_ERR_OP;
72
73 const board_iic_desc_t *d = board_find_iic(h->hw.hi2c.Instance);
74
75 HAL_StatusTypeDef ret = HAL_I2C_DeInit(&h->hw.hi2c);
76 h->initialized = 0;
77 h->last_err = _hal_to_os_err(ret);
78
79 if (d != NULL)
80 {
81 drv_irq_disable((int32_t)d->ev_irqn);
82 drv_irq_disable((int32_t)d->er_irqn);
83 HAL_GPIO_DeInit(d->scl_pin.port, d->scl_pin.pin);
84 HAL_GPIO_DeInit(d->sda_pin.port, d->sda_pin.pin);
85 }
86
87 return h->last_err;
88}

90static int32_t stm32_iic_transmit(drv_iic_handle_t *h,
91 uint16_t dev_addr,
92 uint8_t reg_addr,
93 uint8_t use_reg,
94 const uint8_t *data,
95 uint16_t len,
96 uint32_t timeout_ms)
97{
98 if (h == NULL || !h->initialized || data == NULL)
99 return OS_ERR_OP;
100
101 HAL_StatusTypeDef ret;
102 uint16_t shifted = (uint16_t)(dev_addr << 1);
103
104 if (use_reg)
105 ret = HAL_I2C_Mem_Write(&h->hw.hi2c, shifted, reg_addr,
106 I2C_MEMADD_SIZE_8BIT, (uint8_t *)data,
107 len, timeout_ms);
108 else
109 ret = HAL_I2C_Master_Transmit(&h->hw.hi2c, shifted,
110 (uint8_t *)data, len, timeout_ms);
111
112 h->last_err = _hal_to_os_err(ret);
113 return h->last_err;
114}

244/*
245 * hal_iic_stm32_ev_irq_handler — I2C event IRQ, dispatches completion events.
246 *
247 * Captures HAL state before and after calling HAL_I2C_EV_IRQHandler. When
248 * the state transitions to READY the transfer is done; the previous state
249 * (BUSY_TX vs BUSY_RX) selects which IRQ ID to dispatch. Both Master and
250 * Mem transfer variants share the same BUSY_TX / BUSY_RX states so no
251 * separate MemTxCplt / MemRxCplt path is needed.
252 */
253void hal_iic_stm32_ev_irq_handler(I2C_TypeDef *instance)
254{
255 for (uint8_t id = 0; id < NO_OF_IIC; id++)
256 {
257 drv_iic_handle_t *h = drv_iic_get_handle(id);
258 if (h == NULL || !h->initialized || h->hw.hi2c.Instance != instance)
259 continue;
260
261 HAL_I2C_StateTypeDef prev = HAL_I2C_GetState(&h->hw.hi2c);
262 HAL_I2C_EV_IRQHandler(&h->hw.hi2c);
263 HAL_I2C_StateTypeDef next = HAL_I2C_GetState(&h->hw.hi2c);
264
265 if (next == HAL_I2C_STATE_READY && prev != HAL_I2C_STATE_READY)
266 {
267 h->last_err = OS_ERR_NONE;
268 BaseType_t hpt = pdFALSE;
269 if (prev == HAL_I2C_STATE_BUSY_TX)
270 drv_irq_dispatch_from_isr(IRQ_ID_I2C_TX_DONE(id), NULL, &hpt);
271 else
272 drv_irq_dispatch_from_isr(IRQ_ID_I2C_RX_DONE(id), NULL, &hpt);
273 portYIELD_FROM_ISR(hpt);
274 }
275 return;
276 }
277}

279/*
280 * hal_iic_stm32_er_irq_handler — I2C error IRQ.
281 *
282 * Calls HAL_I2C_ER_IRQHandler (which resets the state machine to READY), then
283 * reads the error code and dispatches IRQ_ID_I2C_ERROR directly — no
284 * HAL_I2C_ErrorCallback weak override needed.
285 */
286void hal_iic_stm32_er_irq_handler(I2C_TypeDef *instance)
287{
288 for (uint8_t id = 0; id < NO_OF_IIC; id++)
289 {
290 drv_iic_handle_t *h = drv_iic_get_handle(id);
291 if (h == NULL || !h->initialized || h->hw.hi2c.Instance != instance)
292 continue;
293
294 HAL_I2C_ER_IRQHandler(&h->hw.hi2c);
295
296 uint32_t err = HAL_I2C_GetError(&h->hw.hi2c);
297 h->last_err = OS_ERR_OP;
298 BaseType_t hpt = pdFALSE;
299 drv_irq_dispatch_from_isr(IRQ_ID_I2C_ERROR(id), &err, &hpt);
300 portYIELD_FROM_ISR(hpt);
301 return;
302 }
303}
