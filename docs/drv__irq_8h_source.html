1/*
2 * drv_irq.h — Vendor-agnostic IRQ abstraction
3 *
4 * This file is part of FreeRTOS-OS Project.
5 *
6 * Two orthogonal concerns live here:
7 *
8 * 1. NVIC control (vendor-specific, compiled only for the active MCU family):
9 * drv_irq_enable / drv_irq_disable / drv_irq_set_priority
10 *
11 * 2. Software IRQ registration and dispatch (vendor-agnostic):
12 * drv_irq_register / drv_irq_dispatch / drv_irq_dispatch_from_isr
13 *
14 * HAL callbacks call drv_irq_dispatch_from_isr() with the IRQ ID.
15 * Drivers and management tasks call drv_irq_register() at startup to
16 * subscribe to events. Neither layer touches irq_notify.h directly.
17 */
18

54/* ── Software IRQ registration and dispatch (vendor-agnostic) ───────────── */
55
56/**
57 * @brief Register a callback for a software IRQ ID.
58 * Thin wrapper over irq_register() — upper layers include only
59 * this header and never touch irq_notify.h directly.
60 * @return 0 on success, negative on error (subscriber table full, bad id).
61 */
62int32_t drv_irq_register(irq_id_t id, irq_notify_cb_t cb, void *arg);
63
64/**
65 * @brief Dispatch an IRQ event from task context.
66 * Calls every callback registered for @p id.
67 */
68void drv_irq_dispatch(irq_id_t id, void *data);
69
70/**
71 * @brief Dispatch an IRQ event from ISR context.
72 * Each registered callback receives @p pxHPT so it can accumulate
73 * the FreeRTOS yield flag. Caller must call
74 * portYIELD_FROM_ISR(*pxHPT) after this returns.
75 */
76void drv_irq_dispatch_from_isr(irq_id_t id, void *data, BaseType_t *pxHPT);
