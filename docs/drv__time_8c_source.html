2 * drv_time.c — Generic timer / systick driver
3 *
4 * This file is part of FreeRTOS-OS Project.

6 * Systick utilities (drv_time_get_ticks / drv_time_delay_ms) are always
7 * available and call the vendor HAL directly — no handle needed.
8 *
9 * Per-instance timer support (encoder counters, PWM, etc.) follows the same

42int32_t drv_timer_register(uint8_t dev_id, const drv_timer_hal_ops_t *ops)
43{
44 if (dev_id >= NO_OF_TIMER || ops == NULL)
45 return OS_ERR_OP;
46
47 drv_timer_handle_t *h = &_timer_handles[dev_id];
48 h->dev_id = dev_id;
49 h->ops = ops;
50 h->initialized = 0;
51 h->last_err = OS_ERR_NONE;
52
53 int32_t err = h->ops->hw_init(h);
54 if (err != OS_ERR_NONE)
55 return err;
56
57 h->initialized = 1;
58 return OS_ERR_NONE;
59}

68uint32_t drv_timer_get_counter(uint8_t dev_id)
69{
70 if (dev_id >= NO_OF_TIMER)
71 return 0;
72
73 drv_timer_handle_t *h = &_timer_handles[dev_id];
74 if (!h->initialized || h->ops == NULL || h->ops->get_counter == NULL)
75 return 0;
76
77 return h->ops->get_counter(h);
78}

87 return;
88
89 h->ops->set_counter(h, val);
