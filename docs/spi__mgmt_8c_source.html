1/*
2 * spi_mgmt.c — SPI management service thread
3 *
4 * This file is part of FreeRTOS-OS Project.
5 *
6 * All transfers use interrupt-driven HAL ops (transmit_it / receive_it /
7 * transfer_it). Completion is signalled from the ISR callback via the generic
8 * IRQ notification framework:
9 *
10 * mgmt thread ISR (HAL_SPI_*CpltCallback)
11 * ────────── ───────────────────────────
12 * h->notify_task = currentTask()
13 * ops->transfer_it(...) ──────────► irq_notify_from_isr(IRQ_ID_SPI_TXRX_DONE)
14 * ulTaskNotifyTake(...) ◄────────── vTaskNotifyGiveFromISR(h->notify_task)
15 * result = h->last_err
16 */

55static void spi_mgmt_thread(void *arg)
56{
57 (void)arg;
58
59 os_thread_delay(TIME_OFFSET_SPI_MANAGEMENT);
60
61 /* Register all SPI buses described in the board configuration */
62 {
63 const drv_spi_hal_ops_t *ops =
64#if (CONFIG_DEVICE_VARIANT == MCU_VAR_STM) && defined(HAL_SPI_MODULE_ENABLED)
65 hal_spi_stm32_get_ops();
66#else
67 NULL;
68#endif
69 if (ops != NULL)
70 {
71 const board_config_t *bc = board_get_config();
72 for (uint8_t i = 0; i < bc->spi_count; i++)
73 {
74 const board_spi_desc_t *d = &bc->spi_table[i];
75 drv_spi_handle_t *h = drv_spi_get_handle(d->dev_id);
76
77#if (CONFIG_DEVICE_VARIANT == MCU_VAR_STM) && defined(HAL_SPI_MODULE_ENABLED)
78 hal_spi_stm32_set_config(h,
79 d->instance,
80 d->mode,
81 d->direction,
82 d->data_size,
83 d->clk_polarity,
84 d->clk_phase,
85 d->nss,
86 d->baud_prescaler,
87 d->bit_order);
88#endif
89 int32_t err = drv_spi_register(d->dev_id, ops, 0, 10);
90 (void)err;
91
92 /* Subscribe for IT completion and error notifications */
93 irq_register(IRQ_ID_SPI_TX_DONE(d->dev_id), _spi_done_cb, h);
94 irq_register(IRQ_ID_SPI_RX_DONE(d->dev_id), _spi_done_cb, h);
95 irq_register(IRQ_ID_SPI_TXRX_DONE(d->dev_id), _spi_done_cb, h);
96 irq_register(IRQ_ID_SPI_ERROR(d->dev_id), _spi_err_cb, h);
97 }
98 }
99 }
100
101 spi_mgmt_msg_t msg;
102
103 for (;;)
104 {
105 if (xQueueReceive(_mgmt_queue, &msg, portMAX_DELAY) != pdTRUE)
106 continue;
107
108 drv_spi_handle_t *h = drv_spi_get_handle(msg.bus_id);
109 if (h == NULL || h->ops == NULL)
110 goto notify;
111
112 int32_t result = OS_ERR_OP;
113
114 switch (msg.cmd)
115 {
116 case SPI_MGMT_CMD_TRANSMIT:
117 if (h->ops->transmit_it != NULL)
118 {
119 h->notify_task = xTaskGetCurrentTaskHandle();
120 result = h->ops->transmit_it(h, msg.tx_data, msg.len);
121 if (result == OS_ERR_NONE)
122 ulTaskNotifyTake(pdTRUE, pdMS_TO_TICKS(h->timeout_ms));
123 result = h->last_err;
124 h->notify_task = NULL;
125 }
126 else
127 {
128 result = h->ops->transmit(h, msg.tx_data, msg.len,
129 h->timeout_ms);
130 }
131 break;
132
133 case SPI_MGMT_CMD_RECEIVE:
134 if (h->ops->receive_it != NULL)
135 {
136 h->notify_task = xTaskGetCurrentTaskHandle();
137 result = h->ops->receive_it(h, msg.rx_data, msg.len);
138 if (result == OS_ERR_NONE)
139 ulTaskNotifyTake(pdTRUE, pdMS_TO_TICKS(h->timeout_ms));
140 result = h->last_err;
141 h->notify_task = NULL;
142 }
143 else
144 {
145 result = h->ops->receive(h, msg.rx_data, msg.len,
146 h->timeout_ms);
147 }
148 break;
149
150 case SPI_MGMT_CMD_TRANSFER:
151 if (h->ops->transfer_it != NULL)
152 {
153 h->notify_task = xTaskGetCurrentTaskHandle();
154 result = h->ops->transfer_it(h, msg.tx_data, msg.rx_data,
155 msg.len);
156 if (result == OS_ERR_NONE)
157 ulTaskNotifyTake(pdTRUE, pdMS_TO_TICKS(h->timeout_ms));
158 result = h->last_err;
159 h->notify_task = NULL;
160 }
161 else
162 {
163 result = h->ops->transfer(h, msg.tx_data, msg.rx_data,
164 msg.len, h->timeout_ms);
165 }
166 break;
167
168 case SPI_MGMT_CMD_REINIT:
169 h->ops->hw_deinit(h);
170 result = h->ops->hw_init(h);
171 break;
172
173 default:
174 break;
175 }
176
177 if (msg.result_code != NULL)
178 *msg.result_code = result;
179
180notify:
181 if (msg.result_notify != NULL)
182 xTaskNotifyGive(msg.result_notify);
183 }
184}

186int32_t spi_mgmt_start(void)
187{
188 _mgmt_queue = xQueueCreate(SPI_MGMT_QUEUE_DEPTH, sizeof(spi_mgmt_msg_t));
189 if (_mgmt_queue == NULL)
190 return OS_ERR_MEM_OF;
191
192 int32_t tid = os_thread_create(spi_mgmt_thread,
193 "spi_mgmt",
194 PROC_SERVICE_SPI_MGMT_STACK_SIZE,
195 PROC_SERVICE_SPI_MGMT_PRIORITY,
196 NULL);
197 return (tid >= 0) ? OS_ERR_NONE : OS_ERR_OP;
198}

205int32_t spi_mgmt_sync_transfer(uint8_t bus_id,
206 const uint8_t *tx, uint8_t *rx,
207 uint16_t len, uint32_t timeout_ms)
208{
209 if (_mgmt_queue == NULL)
210 return OS_ERR_OP;
211
212 int32_t result = OS_ERR_OP;
213
214 spi_mgmt_msg_t msg = {
215 .cmd = SPI_MGMT_CMD_TRANSFER,
216 .bus_id = bus_id,
217 .tx_data = tx,
218 .rx_data = rx,
219 .len = len,
220 .result_notify = xTaskGetCurrentTaskHandle(),
221 .result_code = &result,
222 };
223
224 if (xQueueSend(_mgmt_queue, &msg, pdMS_TO_TICKS(timeout_ms)) != pdTRUE)
225 return OS_ERR_OP;
226
227 ulTaskNotifyTake(pdTRUE, pdMS_TO_TICKS(timeout_ms));
228 return result;
229}
