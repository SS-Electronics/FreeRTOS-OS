1/*
2 * syscalls.c — Newlib syscall stubs + printk (bare-metal ARM)
3 *
4 * This file is part of FreeRTOS-OS Project.
5 *
6 * Two responsibilities:
7 *
8 * 1. Newlib low-level I/O hooks required by the C library on a bare-metal
9 * target. _write / _read are weak so board-level code can override them.
10 * __io_putchar / __io_getchar are the byte-level hooks used by _write/_read.
11 *
12 * 2. printk — timestamped formatted string output routed through the UART TX
13 * ring buffer. printk_init() must be called after ipc_queues_init() so the
14 * ring-buffer handle is valid.
15 */

138void printk_init(void)
139{
140#ifdef COMM_PRINTK_HW_ID
141 if (COMM_PRINTK_HW_ID < NO_OF_UART)
142 _printk_rb = (struct ringbuffer *)
143 ipc_mqueue_get_handle(global_uart_tx_mqueue_list[COMM_PRINTK_HW_ID]);
144#else
145# error "printk: COMM_PRINTK_HW_ID not defined — set it in conf_os.h"
146#endif
147}

149int32_t printk(const char *fmt, ...)
150{
151#if (NO_OF_UART > 0)
152 if (!_printk_enabled)
153 return 0;
154
155 va_list args;
156 int ts_len, msg_len, i;
157
158 va_start(args, fmt);
159
160 ATOMIC_ENTER_CRITICAL();
161
162 ts_len = snprintf(_time_buf, sizeof(_time_buf), "[%lu] ",
163 (unsigned long)drv_time_get_ticks());
164 for (i = 0; i < ts_len; i++)
165 if (_printk_rb) ringbuffer_putchar(_printk_rb, (uint8_t)_time_buf[i]);
166
167 msg_len = vsnprintf(_msg_buf, sizeof(_msg_buf), fmt, args);
168 if (msg_len > 0)
169 for (i = 0; i < msg_len; i++)
170 if (_printk_rb) ringbuffer_putchar(_printk_rb, (uint8_t)_msg_buf[i]);
171
172 ATOMIC_EXIT_CRITICAL();
173
174 va_end(args);
175
176 drv_uart_tx_kick(COMM_PRINTK_HW_ID);
177 return msg_len;
178#else
179 (void)fmt;
180 return 0;
181#endif
182}
