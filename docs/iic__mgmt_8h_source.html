1/*
2 * iic_mgmt.h — I2C management service thread
3 *
4 * This file is part of FreeRTOS-OS Project.
5 *
6 * The I2C management thread serialises all I2C bus transactions through a
7 * single FreeRTOS queue, ensuring that concurrent tasks do not corrupt bus
8 * state. Callers post iic_mgmt_msg_t requests and optionally provide a
9 * notification handle to receive the result asynchronously.
10 */
11

25/* ── Message types ────────────────────────────────────────────────────── */

27typedef enum {
28 IIC_MGMT_CMD_TRANSMIT = 0, /**< Write bytes to a device */
29 IIC_MGMT_CMD_RECEIVE, /**< Read bytes from a device */
30 IIC_MGMT_CMD_PROBE, /**< Check if a device responds */
31 IIC_MGMT_CMD_REINIT, /**< Re-initialise the bus */
32} iic_mgmt_cmd_t;
33
34/**
35 * @struct iic_mgmt_msg_t
36 * @brief Message posted to the I2C management thread queue.
37 *
38 * @note result_notify — if non-NULL the management thread calls
39 * xTaskNotifyGive() on this handle after completing the operation,
40 * and stores the return code in *result_code (if non-NULL).
41 * The caller may then ulTaskNotifyTake() to synchronise.
42 */
43typedef struct {
44 iic_mgmt_cmd_t cmd;
45 uint8_t bus_id; /**< I2C bus index (0-based) */
46 uint16_t dev_addr; /**< 7-bit I2C device address */
47 uint8_t reg_addr; /**< Register address (if use_reg) */
48 uint8_t use_reg; /**< Non-zero to perform a reg access */
49 uint8_t *data; /**< TX data (TRANSMIT) / RX buf (RX) */
50 uint16_t len; /**< Byte count */
51 TaskHandle_t result_notify; /**< Task to notify on completion */
52 int32_t *result_code; /**< Pointer to store the return code */
53} iic_mgmt_msg_t;
