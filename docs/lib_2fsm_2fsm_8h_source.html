74/** @brief Event identifier — define your own enum and cast to this. */

77/**
78 * @brief FSM event object.
79 *
80 * Allocated on the heap by fsm_dispatch() / fsm_dispatch_from_isr()
81 * and automatically freed by fsm_run() after the transition completes.
82 *
83 * 'data' is a pointer to caller-managed storage; the FSM never touches
84 * or frees it.
85 */
86typedef struct fsm_event
87{
88 fsm_event_id_t id; /**< Event identifier */
89 void *data; /**< Optional payload (caller-managed) */
90 struct fsm_event *next; /**< Intrusive linked-list linkage */
91} fsm_event_t;

98/**
99 * @brief Transition action callback.
100 *
101 * Executed between the exit handler(s) of the source state and the
102 * entry handler(s) of the target state.
103 *
104 * @param event Triggering event (read-only during action).
105 * @param fsm FSM context — access application data via fsm->user_data.
106 */
107typedef void (*fsm_action_fn)(const fsm_event_t *event,

111 * @brief State entry / exit callback.
112 *
113 * @param state The state being entered or exited.
114 * @param fsm FSM context.

116typedef void (*fsm_state_fn)(struct fsm_state *state,
117 struct fsm_ctx *fsm);

119/**
120 * @brief Lock / unlock hook for queue protection.
121 *
122 * Provide platform-specific implementations when FSM_THREAD_SAFE == 1.
123 *
124 * Examples:
125 * Bare-metal ISR-safe: __disable_irq() / __enable_irq()
126 * FreeRTOS task: xSemaphoreTake / xSemaphoreGive
127 * FreeRTOS task+ISR: taskENTER_CRITICAL / taskEXIT_CRITICAL
128 * CMSIS-RTOS: osMutexAcquire / osMutexRelease
129 */
130typedef void (*fsm_lock_fn)(void);
131typedef void (*fsm_unlock_fn)(void);

138/**
139 * @brief A single state in the FSM / HSM.
140 *
141 * Set 'parent' to enable hierarchical (nested) states — the FSM will
142 * automatically call exit/entry handlers in the correct top-down order.
143 *
144 * Static initialiser example:
145 * @code
146 * static fsm_state_t st_idle = {
147 * .id = ST_IDLE,
148 * .parent = NULL,
149 * .on_entry = idle_entry,
150 * .on_exit = NULL,
151 * };
152 * @endcode
153 */
154typedef struct fsm_state
155{
156 uint32_t id; /**< Unique state identifier */
157 struct fsm_state *parent; /**< Parent state (HSM) or NULL */
158 fsm_state_fn on_entry; /**< Entry handler or NULL */
159 fsm_state_fn on_exit; /**< Exit handler or NULL */
160} fsm_state_t;
161
162
163/* ================================================================== */
164/* Transition table row */
165/* ================================================================== */
166
167/**
168 * @brief One row of the transition table.
169 *
170 * Build a const array of these and pass it to fsm_init().
171 *
172 * @code
173 * static const fsm_transition_t tbl[] = {
174 * // { from, event, to, action }
175 * { &st_idle, EVT_START, &st_run, act_start },
176 * { &st_run, EVT_DONE, &st_idle, act_cleanup },
177 * { &st_run, EVT_ERROR, &st_err, act_log },
178 * };
179 * @endcode
180 */
181typedef struct
182{
183 fsm_state_t *from; /**< Source state */
184 fsm_event_id_t event; /**< Trigger event id */
185 fsm_state_t *to; /**< Target state */
186 fsm_action_fn action; /**< Transition action or NULL */
187} fsm_transition_t;

194/**
195 * @brief Runtime context of one FSM instance.
196 *
197 * Embed one per logical state machine; may be static or heap-allocated.
198 * Initialise with fsm_init() before use.
199 */
200typedef struct fsm_ctx
201{
202 fsm_state_t *current; /**< Active state */
203
204 const fsm_transition_t *table; /**< Transition table */
205 size_t table_len; /**< Number of rows */
206
207 void *user_data; /**< Caller context */
208
209 /* Internal FIFO event queue (intrusive linked-list) */
210 fsm_event_t *queue_head; /**< Front of queue */
211 fsm_event_t *queue_tail; /**< Back of queue */
212
213 volatile int32_t running; /**< 1=active, 0=stopped */
214
215#if (FSM_THREAD_SAFE == 1)
216 fsm_lock_fn lock; /**< Task-context lock */
217 fsm_unlock_fn unlock; /**< Task-context unlock */
218 fsm_lock_fn lock_isr; /**< ISR-context lock */
219 fsm_unlock_fn unlock_isr; /**< ISR-context unlock */
220#endif
221
222} fsm_ctx_t;

229/**
230 * @brief Initialise an FSM instance.
231 *
232 * Clears the internal queue, sets the initial state, and fires its
233 * on_entry handler.
234 *
235 * @param fsm FSM context (caller-allocated; may be static).
236 * @param initial First active state.
237 * @param table Transition table array (may be const / ROM).
238 * @param table_len Number of rows in 'table'.
239 * @param user_data Arbitrary context passed to every callback.
240 *
241 * @return FSM_OK or FSM_ERR_NULL.
242 */
243int fsm_init(fsm_ctx_t *fsm,
244 fsm_state_t *initial,
245 const fsm_transition_t *table,
246 size_t table_len,
247 void *user_data);

285/**
286 * @brief Post an event from task / thread context.
287 *
288 * Allocates an event on the heap and appends it to the internal queue.
289 * The event is freed automatically by fsm_run() after processing.
290 *
291 * @param fsm FSM context.
292 * @param id Event identifier.
293 * @param data Optional payload pointer (caller manages lifetime).
294 *
295 * @return FSM_OK, FSM_ERR_NULL, or FSM_ERR_ALLOC.
296 */
297int fsm_dispatch(fsm_ctx_t *fsm,
298 fsm_event_id_t id,
299 void *data);

302/**
303 * @brief Post an event from an ISR.
304 *
305 * Identical to fsm_dispatch() but uses the ISR lock/unlock hooks
306 * instead of the task hooks.
307 *
308 * @note malloc() is called internally. Ensure your heap allocator
309 * is interrupt-safe (e.g. FreeRTOS heap_4 with
310 * configUSE_NEWLIB_REENTRANT or a custom ISR-safe allocator).
311 * Alternatively call from a deferred interrupt task.
312 *
313 * @param fsm FSM context.
314 * @param id Event identifier.
315 * @param data Optional payload pointer.
316 *
317 * @return FSM_OK, FSM_ERR_NULL, or FSM_ERR_ALLOC.
318 */
319int fsm_dispatch_from_isr(fsm_ctx_t *fsm,
320 fsm_event_id_t id,
321 void *data);

324/**
325 * @brief Process all pending events — call this in your task loop.
326 *
327 * Drains the internal queue: for each event it finds the matching
328 * transition, runs exit → action → entry handlers, updates the active
329 * state, and frees the event. Returns when the queue is empty.
330 *
331 * Events with no matching transition for the current state are
332 * silently discarded.
333 *
334 * Typical patterns:
335 * @code
336 * // Bare-metal super-loop
337 * for (;;) { fsm_run(&fsm); }
338 *
339 * // RTOS task — unblock via semaphore/notification, then drain
340 * for (;;) {
341 * ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
342 * fsm_run(&fsm);
343 * }
344 * @endcode
345 *
346 * @param fsm FSM context.
347 */
348void fsm_run(fsm_ctx_t *fsm);
