46/* Internal: event queue (intrusive singly-linked FIFO) */

49/**
50 * @brief Append an event to the tail of the internal queue.
51 *
52 * Caller must hold the appropriate lock before calling.
53 */
54static void queue_push(fsm_ctx_t *fsm, fsm_event_t *event)
55{
56 event->next = NULL;
57
58 if (fsm->queue_tail == NULL)
59 {
60 /* Queue was empty */
61 fsm->queue_head = event;
62 fsm->queue_tail = event;
63 }
64 else
65 {
66 fsm->queue_tail->next = event;
67 fsm->queue_tail = event;
68 }
69}
70
71
72/**
73 * @brief Remove and return the event at the head of the queue.
74 *
75 * Returns NULL if the queue is empty.
76 * Caller must hold the appropriate lock before calling.
77 */
78static fsm_event_t *queue_pop(fsm_ctx_t *fsm)
79{
80 fsm_event_t *e = fsm->queue_head;
81
82 if (e != NULL)
83 {
84 fsm->queue_head = e->next;
85
86 if (fsm->queue_head == NULL)
87 {
88 fsm->queue_tail = NULL;
89 }
90
91 e->next = NULL;
92 }
93
94 return e;
95}

98/* ================================================================== */
99/* Internal: HSM helper — hierarchy traversal */
100/* ================================================================== */

102/**
103 * @brief Walk up the parent chain calling on_exit until 'stop'.
104 *
105 * 'stop' is NOT exited (it is the LCA — common ancestor of source
106 * and target states, or NULL to exit all the way to the root).
107 */
108static void fsm_exit_up_to(fsm_ctx_t *fsm, const fsm_state_t *stop)
109{
110 fsm_state_t *s = fsm->current;
111
112 while (s != NULL && s != stop)
113 {
114 if (s->on_exit != NULL)
115 {
116 s->on_exit(s, fsm);
117 }
118 s = s->parent;
119 }
120}

123/**
124 * @brief Walk down from 'ancestor' to 'target' calling on_entry.
125 *
126 * Entry handlers are fired top-down so that outer (parent) states
127 * are always entered before inner (child) states.
128 */
129static void fsm_enter_down_from(fsm_ctx_t *fsm,
130 const fsm_state_t *ancestor,
131 fsm_state_t *target)
132{
133 /* Build the path from target up to (not including) ancestor */
134 fsm_state_t *stack[FSM_HIERARCHY_DEPTH_MAX];
135 uint32_t depth = 0U;
136 fsm_state_t *s = target;
137
138 while (s != NULL && s != ancestor)
139 {
140 if (depth < FSM_HIERARCHY_DEPTH_MAX)
141 {
142 stack[depth++] = s;
143 }
144 s = s->parent;
145 }
146
147 /* Replay entry handlers from ancestor downwards */
148 while (depth > 0U)
149 {
150 --depth;
151 if (stack[depth]->on_entry != NULL)
152 {
153 stack[depth]->on_entry(stack[depth], fsm);
154 }
155 }
156}

159/**
160 * @brief Find the Lowest Common Ancestor (LCA) of states a and b.
161 *
162 * Walks the parent chain of 'a' and for each ancestor checks whether
163 * it also appears in the ancestry of 'b'.
164 *
165 * Returns NULL when there is no common ancestor (flat, no parents).
166 */
167static fsm_state_t *fsm_lca(const fsm_state_t *a, const fsm_state_t *b)
168{
169 const fsm_state_t *pa;
170 const fsm_state_t *pb;
171
172 for (pa = a; pa != NULL; pa = pa->parent)
173 {
174 for (pb = b; pb != NULL; pb = pb->parent)
175 {
176 if (pa == pb)
177 {
178 return (fsm_state_t *)pa;
179 }
180 }
181 }
182
183 return NULL;
184}

212int fsm_init(fsm_ctx_t *fsm,
213 fsm_state_t *initial,
214 const fsm_transition_t *table,
215 size_t table_len,
216 void *user_data)
217{
218 if (fsm == NULL || initial == NULL || table == NULL)
219 {
220 return FSM_ERR_NULL;
221 }
222
223 fsm->current = initial;
224 fsm->table = table;
225 fsm->table_len = table_len;
226 fsm->user_data = user_data;
227 fsm->queue_head = NULL;
228 fsm->queue_tail = NULL;
229 fsm->running = 1;
230
231#if (FSM_THREAD_SAFE == 1)
232 fsm->lock = NULL;
233 fsm->unlock = NULL;
234 fsm->lock_isr = NULL;
235 fsm->unlock_isr = NULL;
236#endif
237
238 /* Fire entry handler of the initial state */
239 if (initial->on_entry != NULL)
240 {
241 initial->on_entry(initial, fsm);
242 }
243
244 return FSM_OK;
245}

277int fsm_dispatch(fsm_ctx_t *fsm, fsm_event_id_t id, void *data)
278{
279 if (fsm == NULL)
280 {
281 return FSM_ERR_NULL;
282 }
283
284 fsm_event_t *e = fsm_event_alloc(id, data);
285
286 if (e == NULL)
287 {
288 return FSM_ERR_ALLOC;
289 }
290
291 FSM_LOCK(fsm);
292 queue_push(fsm, e);
293 FSM_UNLOCK(fsm);
294
295 return FSM_OK;
296}

321void fsm_run(fsm_ctx_t *fsm)
322{
323 if (fsm == NULL || fsm->running == 0)
324 {
325 return;
326 }
327
328 fsm_event_t *event;
329
330 /* Drain the entire queue before returning */
331 for (;;)
332 {
333 /* Pop under lock to avoid race with concurrent dispatch */
334 FSM_LOCK(fsm);
335 event = queue_pop(fsm);
336 FSM_UNLOCK(fsm);
337
338 if (event == NULL)
339 {
340 break; /* Queue empty — done for this cycle */
341 }
342
343 /* Search transition table for (current_state, event_id) */
344 for (size_t i = 0U; i < fsm->table_len; ++i)
345 {
346 if (fsm->table[i].from == fsm->current &&
347 fsm->table[i].event == event->id)
348 {
349 fsm_state_t *target = fsm->table[i].to;
350
351 /* Find LCA for correct HSM entry/exit ordering */
352 fsm_state_t *lca = fsm_lca(fsm->current, target);
353
354 /* 1. Exit current state up to (not including) LCA */
355 fsm_exit_up_to(fsm, lca);
356
357 /* 2. Execute transition action */
358 if (fsm->table[i].action != NULL)
359 {
360 fsm->table[i].action(event, fsm);
361 }
362
363 /* 3. Enter target state down from LCA */
364 fsm_enter_down_from(fsm, lca, target);
365
366 /* 4. Commit new state */
367 fsm->current = target;
368
369 break;
370 }
371 }
372
373 /* Events with no matching transition are silently discarded */
374 free(event);
375 }
376}
