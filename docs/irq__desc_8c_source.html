1/*
2 * irq_desc.c — Linux-style IRQ descriptor chain for FreeRTOS/bare-metal
3 *
4 * This file is part of FreeRTOS-OS Project.
5 *
6 * Implements the full irq_desc dispatch chain:
7 *
8 * __do_IRQ_from_isr(irq_id, data, pxHPT)
9 * → irq_to_desc(irq_id) — O(1) table lookup
10 * → depth check (disabled?)
11 * → desc->handle_irq(desc, data, pxHPT) — flow handler
12 * → handle_irq_event(desc, data, pxHPT) — walk irqaction chain
13 * → action->handler(irq, data, dev_id, pxHPT)
14 *
15 * Memory:
16 * irqaction nodes are allocated from the FreeRTOS heap via kmaloc/kfree.
17 * Each desc->action_list is a doubly-circular sentinel (mm/list.h).
18 * irq_desc_table is a flat array indexed by irq_id_t.
19 */
20

58/* ── Action chain walker ──────────────────────────────────────────────────── */
59
60void handle_irq_event(struct irq_desc *desc, void *data, BaseType_t *pxHPT)
61{
62 if (desc == NULL)
63 return;
64
65 struct irqaction *action;
66 irqreturn_t res = IRQ_NONE;
67
68 list_for_each_entry(action, &desc->action_list, node)
69 {
70 if (action->handler != NULL)
71 res |= action->handler(desc->irq_data.irq, data,
72 action->dev_id, pxHPT);
73 }
74
75 if (res == IRQ_NONE)
76 {
77 desc->irqs_unhandled++;
78 desc->last_unhandled = desc->irq_count; /* snapshot for debug */
79 }
80}

123 if (chip != NULL && chip->irq_unmask != NULL)
124 chip->irq_unmask(&desc->irq_data);

127/* ── Central dispatch ─────────────────────────────────────────────────────── */
128
129void __do_IRQ_from_isr(irq_id_t irq, void *data, BaseType_t *pxHPT)
130{
131 struct irq_desc *desc = irq_to_desc(irq);
132 if (desc == NULL)
133 return;
134
135 desc->irq_count++;
136 desc->tot_count++;
137
138 if (desc->depth > 0)
139 {
140 desc->irqs_unhandled++;
141 return;
142 }
143
144 if (desc->handle_irq != NULL)
145 desc->handle_irq(desc, data, pxHPT);
146}

168
169int request_irq(irq_id_t irq, irq_handler_t handler,
170 const char *name, void *dev_id)
171{
172 if (irq >= IRQ_ID_TOTAL || handler == NULL)
173 return OS_ERR_OP;
174
175 struct irq_desc *desc = &_irq_desc_table[irq];
176
177 struct irqaction *new_action = _alloc_action();
178 if (new_action == NULL)
179 return OS_ERR_MEM_OF;
180
181 new_action->handler = handler;
182 new_action->dev_id = dev_id;
183 new_action->name = name;
184
185 list_add_tail(&new_action->node, &desc->action_list);
186
187 return OS_ERR_NONE;
188}

190void free_irq(irq_id_t irq, void *dev_id)
191{
192 if (irq >= IRQ_ID_TOTAL)
193 return;
194
195 struct irq_desc *desc = &_irq_desc_table[irq];
196 struct irqaction *cur;
197
198 list_for_each_entry(cur, &desc->action_list, node)
199 {
200 if (cur->dev_id == dev_id)
201 {
202 list_delete(&cur->node);
203 _free_action(cur);
204 return;
205 }
206 }
207}

236void irq_disable(irq_id_t irq)
237{
238 if (irq >= IRQ_ID_TOTAL)
239 return;

242 desc->depth++;
243
244 struct irq_chip *chip = desc->irq_data.chip;
245 if (chip != NULL && chip->irq_disable != NULL)
246 chip->irq_disable(&desc->irq_data);
247 else if (chip != NULL && chip->irq_mask != NULL)

251void irq_enable(irq_id_t irq)
252{
253 if (irq >= IRQ_ID_TOTAL)
254 return;
255
256 struct irq_desc *desc = &_irq_desc_table[irq];
257 if (desc->depth > 0)
258 desc->depth--;
259
260 if (desc->depth == 0)
261 {
262 struct irq_chip *chip = desc->irq_data.chip;
263 if (chip != NULL && chip->irq_enable != NULL)
264 chip->irq_enable(&desc->irq_data);
265 else if (chip != NULL && chip->irq_unmask != NULL)
266 chip->irq_unmask(&desc->irq_data);
267 }
268}

273{
274 for (irq_id_t i = 0; i < IRQ_ID_TOTAL; i++)

276 struct irq_desc *desc = &_irq_desc_table[i];
277
278 desc->irq_common_data.state_use_accessors = 0;
279 desc->irq_common_data.handler_data = NULL;
280
281 desc->irq_data.irq = i;
282 desc->irq_data.hwirq = -1;
283 desc->irq_data.chip = NULL;
284 desc->irq_data.desc = desc;
285 desc->irq_data.chip_data = NULL;
286
287 desc->handle_irq = handle_simple_irq;
288 list_init(&desc->action_list);
289 desc->status_use_accessors = 0;
290 desc->depth = 0;
291 desc->tot_count = 0;
292 desc->irq_count = 0;
293 desc->last_unhandled = 0;
294 desc->irqs_unhandled = 0;
295 desc->name = irq_table_get_name(i);
