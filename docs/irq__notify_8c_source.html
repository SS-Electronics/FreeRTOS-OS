1/*
2 * irq_notify.c — pub/sub adapter over the irq_desc chain
3 *
4 * This file is part of FreeRTOS-OS Project.
5 *
6 * Provides the original irq_notify API (irq_register / irq_notify /
7 * irq_notify_from_isr) as a thin adapter over irq_desc.c.
8 *
9 * The signature mismatch between irq_notify_cb_t (void return) and
10 * irq_handler_t (irqreturn_t return) is resolved by a static trampoline pool:
11 * each irq_register() call allocates a trampoline that wraps the cb+arg pair
12 * and is registered with request_irq() as the irq_handler_t.
13 *
14 * Dispatch path (ISR):
15 * drv_irq_dispatch_from_isr()
16 * → __do_IRQ_from_isr()
17 * → desc->handle_irq() (handle_simple_irq by default)
18 * → handle_irq_event()
19 * → _notify_trampoline() (irq_handler_t)
20 * → irq_notify_cb_t (original subscriber)
21 */
22

27/* ── Trampoline pool ──────────────────────────────────────────────────────── */
28
29typedef struct {
30 irq_notify_cb_t cb;
31 void *arg;
32 uint8_t used;
33} _notify_trampoline_t;
34
35#define _TRAMPOLINE_POOL_SIZE (IRQ_ID_TOTAL * IRQ_NOTIFY_MAX_SUBS)
36
37static _notify_trampoline_t _trampoline_pool[_TRAMPOLINE_POOL_SIZE];
38
39static _notify_trampoline_t *_alloc_trampoline(irq_notify_cb_t cb, void *arg)
40{
41 for (unsigned int i = 0; i < _TRAMPOLINE_POOL_SIZE; i++)
42 {
43 if (!_trampoline_pool[i].used)
44 {
45 _trampoline_pool[i].cb = cb;
46 _trampoline_pool[i].arg = arg;
47 _trampoline_pool[i].used = 1;
48 return &_trampoline_pool[i];
49 }
50 }
51 return NULL;
52}

54/* ── Trampoline handler (irq_handler_t signature) ────────────────────────── */
55
56static irqreturn_t _notify_trampoline(irq_id_t irq,
57 void *data,
58 void *dev_id,
59 BaseType_t *pxHPT)
60{
61 _notify_trampoline_t *t = (_notify_trampoline_t *)dev_id;
62 if (t != NULL && t->cb != NULL)
63 {
64 t->cb(irq, data, t->arg, pxHPT);
65 return IRQ_HANDLED;
66 }
67 return IRQ_NONE;
68}

72int32_t irq_register(irq_id_t id, irq_notify_cb_t cb, void *arg)
73{
74 if (id >= IRQ_ID_TOTAL || cb == NULL)
75 return OS_ERR_OP;
76
77 /* Reject duplicate (id, cb, arg) registrations */
78 for (unsigned int i = 0; i < _TRAMPOLINE_POOL_SIZE; i++)
79 {
80 if (_trampoline_pool[i].used &&
81 _trampoline_pool[i].cb == cb &&
82 _trampoline_pool[i].arg == arg)
83 return OS_ERR_NONE;
84 }
85
86 _notify_trampoline_t *t = _alloc_trampoline(cb, arg);
87 if (t == NULL)
88 return OS_ERR_MEM_OF;
89
90 int32_t err = request_irq(id, _notify_trampoline, NULL, t);
91 if (err != OS_ERR_NONE)
92 {
93 t->used = 0;
94 return err;
95 }
96
97 return OS_ERR_NONE;
98}
