1/*
2 * irq_desc.h — Linux-style IRQ descriptor chain for FreeRTOS/bare-metal
3 *
4 * This file is part of FreeRTOS-OS Project.
5 *
6 * Adapts the Linux IRQ descriptor model to a single-core bare-metal system:
7 * - SMP / percpu / cpumask features removed
8 * - PM sleep / proc-fs / sparse-IRQ fields removed
9 * - raw_spinlock_t → not needed (single-core, ISR entry is atomic)
10 * - wait_queue_head_t → FreeRTOS tasks use direct task-notification instead
11 * - All struct and typedef names are kept faithful to Linux for readability
12 *
13 * Call chain (from hardware to driver):
14 *
15 * CPU vector (startup)
16 * → irq_periph_dispatch_generated.c (USART1_IRQHandler etc.)
17 * → hal_xxx_stm32_irq_handler()
18 * → drv_irq_dispatch_from_isr(irq_id, data, pxHPT)
19 * → __do_IRQ_from_isr(irq_id, data, pxHPT)
20 * → irq_to_desc(irq_id) — O(1) array lookup
21 * → desc->handle_irq(desc, …) — flow handler
22 * → handle_irq_event(desc, …) — walk irqaction chain
23 * → action->handler(irq, data, dev_id, pxHPT)
24 *
25 * Registration (driver init, task context):
26 * request_irq(IRQ_ID_UART_RX(0), my_rx_handler, "uart0_rx", ctx);
27 */
28

91/* ── irq_handler_t — driver ISR signature ────────────────────────────────── */

93/**
94 * @param irq Software IRQ ID that fired.
95 * @param data Event payload (e.g. pointer to received byte, error code).
96 * @param dev_id Opaque context supplied to request_irq().
97 * @param pxHPT FreeRTOS yield accumulator; pass to portYIELD_FROM_ISR() after
98 * the dispatch chain returns. NULL in task context.
99 * @return IRQ_HANDLED or IRQ_NONE.

103 void *dev_id,
