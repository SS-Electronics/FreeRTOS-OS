1/*
2 * hal_timebase_stm32.c — HAL tick/timebase via TIM1 (STM32)
3 *
4 * This file is part of FreeRTOS-OS Project.
5 *
6 * Overrides the default SysTick-based HAL_InitTick() so that TIM1 drives the
7 * 1 ms HAL tick counter, leaving SysTick free for FreeRTOS.
8 *
9 * IRQ dispatch follows the irq_desc/irq_chip pattern used by UART, SPI, I2C.
10 * The NVIC chip (irq_chip_nvic_get()) is bound to IRQ_ID_TIM_TICK in
11 * irq_hw_init_all() with handle_edge_irq as the flow handler.
12 *
13 * Call chain:
14 * TIM1_UP_TIM10_IRQHandler (irq_periph_dispatch_generated.c)
15 * → hal_timebase_stm32_irq_handler()
16 * ├─ __HAL_TIM_CLEAR_FLAG — clears TIM1 update pending flag
17 * ├─ g_ms_ticks++ / HAL_IncTick()— advance global ms counter + HAL tick
18 * └─ __do_IRQ_from_isr(IRQ_ID_TIM_TICK)
19 * → handle_edge_irq → nvic_irq_ack (no-op) → irqaction chain
20 * (optional subscribers)
21 *
22 * The hardware flag clear and tick increments run unconditionally so the
23 * system is fully functional before irq_hw_init_all() populates handle_irq.
24 */

38/* ── HAL_InitTick override ──────────────────────────────────────────────── */
39
40HAL_StatusTypeDef HAL_InitTick(uint32_t TickPriority)
41{
42 RCC_ClkInitTypeDef clkconfig;
43 uint32_t uwTimclock;
44 uint32_t uwPrescalerValue;
45 uint32_t pFLatency;
46 HAL_StatusTypeDef status;
47
48 drv_rcc_periph_clk_en(DRV_RCC_PERIPH_TIM1);
49
50 HAL_RCC_GetClockConfig(&clkconfig, &pFLatency);
51 uwTimclock = 2U * HAL_RCC_GetPCLK2Freq();
52 uwPrescalerValue = (uwTimclock / 1000000U) - 1U;
53
54 _htim1.Instance = TIM1;
55 _htim1.Init.Period = (1000000U / 1000U) - 1U;
56 _htim1.Init.Prescaler = uwPrescalerValue;
57 _htim1.Init.ClockDivision = 0;
58 _htim1.Init.CounterMode = TIM_COUNTERMODE_UP;
59 _htim1.Init.AutoReloadPreload = TIM_AUTORELOAD_PRELOAD_DISABLE;
60
61 status = HAL_TIM_Base_Init(&_htim1);
62 if (status == HAL_OK)
63 {
64 status = HAL_TIM_Base_Start_IT(&_htim1);
65 if (status == HAL_OK)
66 {
67 HAL_NVIC_EnableIRQ(TIM1_UP_TIM10_IRQn);
68 if (TickPriority < (1UL << __NVIC_PRIO_BITS))
69 {
70 HAL_NVIC_SetPriority(TIM1_UP_TIM10_IRQn, TickPriority, 0U);
71 uwTickPrio = TickPriority;
72 }
73 else
74 {
75 status = HAL_ERROR;
76 }
77 }
78 }
79
80 return status;
81}

95/* ── IRQ handler ────────────────────────────────────────────────────────── */
96
97void hal_timebase_stm32_irq_handler(void)
98{
99 BaseType_t pxHPT = pdFALSE;
100
101 /* Service hardware first — clear pending flag and advance tick counters.
102 * Done unconditionally so the system works before irq_hw_init_all(). */
103 __HAL_TIM_CLEAR_FLAG(&_htim1, TIM_FLAG_UPDATE);
104 g_ms_ticks++;
105 HAL_IncTick();
106
107 /* Notify any irq_desc subscribers (populated after irq_hw_init_all). */
108 struct irq_desc *desc = irq_to_desc(IRQ_ID_TIM_TICK);
109 if (desc != NULL && desc->handle_irq != NULL)
110 __do_IRQ_from_isr(IRQ_ID_TIM_TICK, NULL, &pxHPT);
111
112 portYIELD_FROM_ISR(pxHPT);
113}
