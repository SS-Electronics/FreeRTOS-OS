1/*
2 * ringbuffer.c — FreeRTOS-safe ring buffer (mirror-bit algorithm)
3 *
4 * This file is part of FreeRTOS-OS Project.
5 *
6 * Dynamic memory model:
7 * ringbuffer_create() — kmaloc() for descriptor + data buffer
8 * ringbuffer_destroy() — kfree() for both
9 *
10 * Thread safety:
11 * putchar / putchar_force — ISR-safe via taskENTER_CRITICAL_FROM_ISR /
12 * taskEXIT_CRITICAL_FROM_ISR. Safe to call
13 * from both ISR and task context.
14 * all other ops — task-safe via taskENTER_CRITICAL /
15 * taskEXIT_CRITICAL. Must not be called from
16 * ISR context.
17 * FreeRTOS nests critical sections by reference count so mixing the two
18 * variants on a single-core Cortex-M4 is deadlock-free.
19 */

30static inline enum ringbuffer_state _status(const struct ringbuffer *rb)
31{
32 if (rb->read_index != rb->write_index)
33 return RT_RINGBUFFER_HALFFULL;
34 return (rb->read_mirror == rb->write_mirror)
35 ? RT_RINGBUFFER_EMPTY : RT_RINGBUFFER_FULL;
36}
37
38static inline uint16_t _data_len(const struct ringbuffer *rb)
39{
40 switch (_status(rb)) {
41 case RT_RINGBUFFER_EMPTY: return 0;
42 case RT_RINGBUFFER_FULL: return (uint16_t)rb->buffer_size;
43 default:
44 return (rb->write_index > rb->read_index)
45 ? rb->write_index - rb->read_index
46 : (uint16_t)rb->buffer_size - (rb->read_index - rb->write_index);
47 }
48}

52struct ringbuffer *ringbuffer_create(uint16_t size)
53{
54 if (size == 0)
55 return NULL;
56
57 struct ringbuffer *rb = (struct ringbuffer *)kmaloc(sizeof(*rb));
58 if (rb == NULL)
59 return NULL;
60
61 uint8_t *buf = (uint8_t *)kmaloc(size);
62 if (buf == NULL) {
63 kfree(rb);
64 return NULL;
65 }
66
67 ringbuffer_init(rb, buf, (int16_t)size);
68 return rb;
69}

71void ringbuffer_destroy(struct ringbuffer *rb)
72{
73 if (rb == NULL)
74 return;
75 if (rb->buffer_ptr != NULL)
76 kfree(rb->buffer_ptr);
77 kfree(rb);
78}

127uint32_t ringbuffer_put(struct ringbuffer *rb, const uint8_t *ptr, uint16_t length)
128{
129 taskENTER_CRITICAL();
130
131 uint16_t space = (uint16_t)rb->buffer_size - _data_len(rb);
132 if (space == 0) {
133 taskEXIT_CRITICAL();
134 return 0;
135 }
136 if (space < length)
137 length = space;
138
139 uint16_t tail = (uint16_t)rb->buffer_size - rb->write_index;
140 if (tail > length) {
141 memcpy(&rb->buffer_ptr[rb->write_index], ptr, length);
142 rb->write_index += length;
143 } else {
144 memcpy(&rb->buffer_ptr[rb->write_index], ptr, tail);
145 memcpy(&rb->buffer_ptr[0], ptr + tail, length - tail);
146 rb->write_mirror = (uint16_t)(~rb->write_mirror) & 1U;
147 rb->write_index = length - tail;
148 }
149
150 taskEXIT_CRITICAL();
151 return length;
152}

154uint32_t ringbuffer_put_force(struct ringbuffer *rb, const uint8_t *ptr, uint16_t length)
155{
156 taskENTER_CRITICAL();
157
158 enum ringbuffer_state old = _status(rb);
159 if (length > (uint16_t)rb->buffer_size)
160 length = (uint16_t)rb->buffer_size;
161
162 uint16_t tail = (uint16_t)rb->buffer_size - rb->write_index;
163 if (tail > length) {
164 memcpy(&rb->buffer_ptr[rb->write_index], ptr, length);
165 rb->write_index += length;
166 if (old == RT_RINGBUFFER_FULL)
167 rb->read_index = rb->write_index;
168 } else {
169 memcpy(&rb->buffer_ptr[rb->write_index], ptr, tail);
170 memcpy(&rb->buffer_ptr[0], ptr + tail, length - tail);
171 rb->write_mirror = (uint16_t)(~rb->write_mirror) & 1U;
172 rb->write_index = length - tail;
173 if (old == RT_RINGBUFFER_FULL) {
174 rb->read_mirror = (uint16_t)(~rb->read_mirror) & 1U;
175 rb->read_index = rb->write_index;
176 }
177 }
178
179 taskEXIT_CRITICAL();
180 return length;
181}

185uint32_t ringbuffer_putchar(struct ringbuffer *rb, uint8_t ch)
186{
187 UBaseType_t ux = taskENTER_CRITICAL_FROM_ISR();
188
189 uint32_t ret = 0;
190 if (_data_len(rb) < (uint16_t)rb->buffer_size) {
191 rb->buffer_ptr[rb->write_index] = ch;
192 if (rb->write_index == (uint16_t)(rb->buffer_size - 1)) {
193 rb->write_mirror = (uint16_t)(~rb->write_mirror) & 1U;
194 rb->write_index = 0;
195 } else {
196 rb->write_index++;
197 }
198 ret = 1;
199 }
200
201 taskEXIT_CRITICAL_FROM_ISR(ux);
202 return ret;
203}

231uint32_t ringbuffer_get(struct ringbuffer *rb, uint8_t *ptr, uint16_t length)
232{
233 taskENTER_CRITICAL();
234
235 uint16_t avail = _data_len(rb);
236 if (avail == 0) {
237 taskEXIT_CRITICAL();
238 return 0;
239 }
240 if (avail < length)
241 length = avail;
242
243 uint16_t tail = (uint16_t)rb->buffer_size - rb->read_index;
244 if (tail > length) {
245 memcpy(ptr, &rb->buffer_ptr[rb->read_index], length);
246 rb->read_index += length;
247 } else {
248 memcpy(ptr, &rb->buffer_ptr[rb->read_index], tail);
249 memcpy(ptr + tail, &rb->buffer_ptr[0], length - tail);
250 rb->read_mirror = (uint16_t)(~rb->read_mirror) & 1U;
251 rb->read_index = length - tail;
252 }
253
254 taskEXIT_CRITICAL();
255 return length;
